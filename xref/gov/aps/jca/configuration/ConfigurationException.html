8   package gov.aps.jca.configuration;

12  /**
13   * Thrown when a <code>Configurable</code> component cannot be configured
14   * properly, or if a value cannot be retrieved properly.
15   *
16   * @author <a href="mailto:fede@apache.org">Federico Barbieri</a>
17   * @author <a href="mailto:stefano@apache.org">Stefano Mazzocchi</a>
18   * @author <a href="mailto:fumagalli@exoffice.com">Pierpaolo Fumagalli</a>
19   */
20  public class ConfigurationException
21      extends Exception
22  {
23      /**
24       * Construct a new <code>ConfigurationException</code> instance.
25       *
26       * @param message The detail message for this exception.
27       */
28      public ConfigurationException( final String message )
29      {
30          this( message, null );
31      }
32  
33      /**
34       * Construct a new <code>ConfigurationException</code> instance.
35       *
36       * @param message The detail message for this exception.
37       * @param throwable the root cause of the exception
38       */
39      public ConfigurationException( final String message, final Throwable throwable )
40      {
41          super( message, throwable );
42      }
43  }
