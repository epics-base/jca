28  /**
29   * A ConfigurationSerializer serializes configurations via SAX2 compliant parser.
30   *
31   * @author <a href="mailto:bloritsch@apache.org">Berin Loritsch</a>
32   * @author <a href="mailto:proyal@apache.org">Peter Royal</a>
33   * @version 1.0
34   */
35  public class DefaultConfigurationSerializer
36  {
37      private SAXTransformerFactory m_tfactory;
38      private Properties m_format = new Properties();
39  
40      /**
41       * Sets the Serializer's use of indentation.  This will cause linefeeds to be added
42       *  after each element, but it does not add any indentation via spaces.
43       * @param indent a <code>boolean</code> value
44       */
45      public void setIndent( boolean indent )
46      {
47          if( indent )
48          {
49              m_format.put( OutputKeys.INDENT, "yes" );
50          }
51          else
52          {
53              m_format.put( OutputKeys.INDENT, "no" );
54          }
55      }
56  
57      /**
58       * Create a ContentHandler for an OutputStream
59       * @param result an <code>OutputStream</code> value
60       * @return contenthandler that goes to specified OutputStream
61       */
62      protected ContentHandler createContentHandler( final Result result )
63      {
64          try
65          {
66              TransformerHandler handler = getTransformerFactory().newTransformerHandler();
67  
68              m_format.put( OutputKeys.METHOD, "xml" );
69              handler.setResult( result );
70              handler.getTransformer().setOutputProperties( m_format );
71  
72              return handler;
73          }
74          catch( final Exception e )
75          {
76              throw new RuntimeException( e.toString() );
77          }
78      }
79  
80      /**
81       * Get the SAXTransformerFactory so we can get a serializer without being
82       * tied to one vendor.
83       * @return a <code>SAXTransformerFactory</code> value
84       */
85      protected SAXTransformerFactory getTransformerFactory()
86      {
87          if( m_tfactory == null )
88          {
89              m_tfactory = (SAXTransformerFactory)TransformerFactory.newInstance();
90          }
91  
92          return m_tfactory;
93      }
94  
95      /**
96       * Serialize the configuration to a ContentHandler
97       * @param handler a <code>ContentHandler</code> to serialize to
98       * @param source a <code>Configuration</code> value
99       * @throws SAXException if an error occurs
100      * @throws ConfigurationException if an error occurs
101      */
102     public void serialize( final ContentHandler handler, final Configuration source )
103         throws SAXException, ConfigurationException
104     {
105         handler.startDocument();
106         serializeElement( handler, new NamespaceSupport(), source );
107         handler.endDocument();
108     }
109 
110     /**
111      * Serialize each Configuration element.  This method is called recursively.
112      * @param handler a <code>ContentHandler</code> to use
113      * @param namespaceSupport a <code>NamespaceSupport</code> to use
114      * @param element a <code>Configuration</code> value
115      * @throws SAXException if an error occurs
116      * @throws ConfigurationException if an error occurs
117      */
118     protected void serializeElement( final ContentHandler handler,
119                                      final NamespaceSupport namespaceSupport,
120                                      final Configuration element )
121         throws SAXException, ConfigurationException
122     {
123         namespaceSupport.pushContext();
124 
125         AttributesImpl attr = new AttributesImpl();
126         String[] attrNames = element.getAttributeNames();
127 
128         if( null != attrNames )
129         {
130             for( int i = 0; i < attrNames.length; i++ )
131             {
132                 attr.addAttribute( "", // namespace URI
133                                    attrNames[ i ], // local name
134                                    attrNames[ i ], // qName
135                                    "CDATA", // type
136                                    element.getAttribute( attrNames[ i ], "" ) // value
137                 );
138             }
139         }
140 
141         final String nsURI = element.getNamespace();
142         String nsPrefix = "";
143 
144         if( element instanceof AbstractConfiguration )
145         {
146             nsPrefix = ( (AbstractConfiguration)element ).getPrefix();
147         }
148         // nsPrefix is guaranteed to be non-null at this point.
149 
150         boolean nsWasDeclared = false;
151 
152         final String existingURI = namespaceSupport.getURI( nsPrefix );
153 
154         // ie, there is no existing URI declared for this prefix or we're
155         // remapping the prefix to a different URI
156         if( existingURI == null || !existingURI.equals( nsURI ) )
157         {
158             nsWasDeclared = true;
159             if( nsPrefix.equals( "" ) && nsURI.equals( "" ) )
160             {
161                 // implicit mapping; don't need to declare
162             }
163             else if( nsPrefix.equals( "" ) )
164             {
165                 // (re)declare the default namespace
166                 attr.addAttribute( "", "xmlns", "xmlns", "CDATA", nsURI );
167             }
168             else
169             {
170                 // (re)declare a mapping from nsPrefix to nsURI
171                 attr.addAttribute( "", "xmlns:" + nsPrefix, "xmlns:" + nsPrefix, "CDATA", nsURI );
172             }
173             handler.startPrefixMapping( nsPrefix, nsURI );
174             namespaceSupport.declarePrefix( nsPrefix, nsURI );
175         }
176 
177         String localName = element.getName();
178         String qName = element.getName();
179         if( nsPrefix == null || nsPrefix.length() == 0 )
180         {
181             qName = localName;
182         }
183         else
184         {
185             qName = nsPrefix + ":" + localName;
186         }
187 
188         handler.startElement( nsURI, localName, qName, attr );
189 
190         String value = element.getValue( null );
191 
192         if( null == value )
193         {
194             Configuration[] children = element.getChildren();
195 
196             for( int i = 0; i < children.length; i++ )
197             {
198                 serializeElement( handler, namespaceSupport, children[ i ] );
199             }
200         }
201         else
202         {
203             handler.characters( value.toCharArray(), 0, value.length() );
204         }
205 
206         handler.endElement( nsURI, localName, qName );
207 
208         if( nsWasDeclared )
209         {
210             handler.endPrefixMapping( nsPrefix );
211         }
212 
213         namespaceSupport.popContext();
214     }

216     /**
217      * Serialize the configuration object to a file using a filename.
218      * @param filename a <code>String</code> value
219      * @param source a <code>Configuration</code> value
220      * @throws SAXException if an error occurs
221      * @throws IOException if an error occurs
222      * @throws ConfigurationException if an error occurs
223      */
224     public void serializeToFile( final String filename, final Configuration source )
225         throws SAXException, IOException, ConfigurationException
226     {
227         serializeToFile( new File( filename ), source );
228     }
229 
230     /**
231      * Serialize the configuration object to a file using a File object.
232      * @param file a <code>File</code> value
233      * @param source a <code>Configuration</code> value
234      * @throws SAXException if an error occurs
235      * @throws IOException if an error occurs
236      * @throws ConfigurationException if an error occurs
237      */
238     public void serializeToFile( final File file, final Configuration source )
239         throws SAXException, IOException, ConfigurationException
240     {
241         OutputStream outputStream = null;
242         try
243         {
244             outputStream = new FileOutputStream( file );
245             serialize( outputStream, source );
246         }
247         finally
248         {
249             if( outputStream != null )
250             {
251                 outputStream.close();
252             }
253         }
254     }
255 
256     /**
257      * Serialize the configuration object to an output stream.
258      * @param outputStream an <code>OutputStream</code> value
259      * @param source a <code>Configuration</code> value
260      * @throws SAXException if an error occurs
261      * @throws IOException if an error occurs
262      * @throws ConfigurationException if an error occurs
263      */
264     public void serialize( final OutputStream outputStream, final Configuration source )
265         throws SAXException, IOException, ConfigurationException
266     {
267         serialize( createContentHandler( new StreamResult( outputStream ) ), source );
268     }
269 
270     /**
271      * Serialize the configuration object to an output stream derived from an
272      * URI.  The URI must be resolveable by the <code>java.net.URL</code> object.
273      * @param uri a <code>String</code> value
274      * @param source a <code>Configuration</code> value
275      * @throws SAXException if an error occurs
276      * @throws IOException if an error occurs
277      * @throws ConfigurationException if an error occurs
278      */
279     public void serialize( final String uri, final Configuration source )
280         throws SAXException, IOException, ConfigurationException
281     {
282         OutputStream outputStream = null;
283         try
284         {
285             outputStream = new URL( uri ).openConnection().getOutputStream();
286             serialize( outputStream, source );
287         }
288         finally
289         {
290             if( outputStream != null )
291             {
292                 outputStream.close();
293             }
294         }
295     }
296 
297     /**
298      * Serialize the configuration object to a string
299      * @param source a <code>Configuration</code> value
300      * @return configuration serialized as a string.
301      * @throws SAXException if an error occurs
302      * @throws ConfigurationException if an error occurs
303      */
304     public String serialize( final Configuration source )
305         throws SAXException, ConfigurationException
306     {
307         final StringWriter writer = new StringWriter();
308 
309         serialize( createContentHandler( new StreamResult( writer ) ), source );
310 
311         return writer.toString();
312     }
