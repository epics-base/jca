7   public abstract class Enum {
8     
9     private final String _name;
10    
11    protected Enum(final String name) {
12      this(name,null);
13    }
14    
15    protected Enum(final String name, final java.util.Map map) {
16      _name=name;
17      if(map!=null) map.put(_name,this);
18    }
19    
20      public final boolean equals( final Object other )  {
21          if( null == other )  {
22              return false;
23          }  else {
24              return other == this || 
25                  ( other.getClass().getName().equals( this.getClass().getName() ) &&
26                  _name.equals( ( (Enum)other )._name ) );
27          }
28      }
29  
30      public int hashCode()  {
31        return _name.hashCode();
32      }
33  
34      public final String getName()  {
35          return _name;
36      }
37  
38      public String toString() {
39          return getClass().getName() + "[" + _name + "]";
40      }
