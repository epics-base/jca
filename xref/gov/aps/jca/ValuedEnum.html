3   public abstract class ValuedEnum extends Enum {
4     private int _value;
5     
6     protected ValuedEnum(String name, int value) {
7       this(name,value,null);
8     }
9     
10    protected ValuedEnum(String name, int value, java.util.Map map) {
11      super(name,map);
12      _value= value;
13    }
14    
15    
16      /**
17       * Get value of enum item.
18       *
19       * @return the enum item's value.
20       */
21      public final int getValue()
22      {
23          return _value;
24      }
25  
26      /**
27       * Test if enum item is equal in value to other enum.
28       *
29       * @param other the other enum
30       * @return true if equal
31       */
32      public final boolean isEqualTo( final ValuedEnum other )
33      {
34          return _value == other._value;
35      }
36  
37      /**
38       * Test if enum item is greater than in value to other enum.
39       *
40       * @param other the other enum
41       * @return true if greater than
42       */
43      public final boolean isGreaterThan( final ValuedEnum other )
44      {
45          return _value > other._value;
46      }
47  
48      /**
49       * Test if enum item is greater than or equal in value to other enum.
50       *
51       * @param other the other enum
52       * @return true if greater than or equal
53       */
54      public final boolean isGreaterThanOrEqual( final ValuedEnum other )
55      {
56          return _value >= other._value;
57      }
58  
59      /**
60       * Test if enum item is less than in value to other enum.
61       *
62       * @param other the other enum
63       * @return true if less than
64       */
65      public final boolean isLessThan( final ValuedEnum other )
66      {
67          return _value < other._value;
68      }
69  
70      /**
71       * Test if enum item is less than or equal in value to other enum.
72       *
73       * @param other the other enum
74       * @return true if less than or equal
75       */
76      public final boolean isLessThanOrEqual( final ValuedEnum other )
77      {
78          return _value <= other._value;
79      }
80  
81      /**
82       * Override toString method to produce human readable description.
83       *
84       * @return String in the form <code>type[name=value]</code>, eg.:
85       * <code>JavaVersion[Java 1.0=100]</code>.
86       */
87      public String toString()
88      {
89          return getClass().getName() + "[" + getName() + "=" + _value + "]";
90      }
91  }
