153   // not-so-nice optimization (using hashmap would be nicer, but not very fast) 
154   static CAStatus[] _cachedTypesByValue;
155   
156   static {
157       // cache all types by value
158       // values are all are all unsigned short
159       int maxValue = 0;
160       for(java.util.Iterator it= _map.values().iterator(); it.hasNext(); ) {
161           CAStatus t= (CAStatus) it.next();
162           if (t.getValue() > maxValue)
163               maxValue = t.getValue(); 
164         }
165       
166       _cachedTypesByValue = new CAStatus[maxValue + 1];
167       for(java.util.Iterator it= _map.values().iterator(); it.hasNext(); ) {
168           CAStatus t= (CAStatus) it.next();
169           if (t.getValue() >= 0)
170               _cachedTypesByValue[t.getValue()] = t;
171         }
172   }

174   private CASeverity _severity;
175   private String _msg;
176   
177   /** Creates a new instance of CAStatus */
178   protected CAStatus(String name, int value, CASeverity severity, String msg) {
179     super(name,value,_map);
180     _severity= severity;
181     _msg=msg;
182   }
183   
184   public boolean isSuccessful() {
185     return _severity.isSuccessful();
186   }
187 
188   public boolean isError() {
189     return _severity.isError();
190   }
191   
192   public boolean isFatal() {
193     return _severity.isFatal();
194   }
195   
196   
197   public CASeverity getSeverity() {
198     return _severity;
199   }
200   
201   public String getMessage() {
202     return _msg;
203   }
204   
205   public String toString() {
206     return getClass().getName() + "[" + getName() + "=" + getValue() + ","+_severity.getName()+"="+_severity.getValue()+"]="+getMessage();
207   }

210   static public CAStatus forName(String name) {
211     return (CAStatus) _map.get(name);

225   static final public CAStatus forValue(final int value ) {
226       if (value >= 0 && value < _cachedTypesByValue.length)
227           return _cachedTypesByValue[value];
228       else
229           return null;
230     }

232   /*  CA Status Code Definitions   */
233 
234   public final static int CA_M_MSG_NO     = 0x0000FFF8;
235   public final static int CA_M_SEVERITY   = 0x00000007;
236   public final static int CA_M_LEVEL      = 0x00000003;
237   public final static int CA_M_SUCCESS    = 0x00000001;
238   public final static int CA_M_ERROR      = 0x00000002;
239   public final static int CA_M_SEVERE     = 0x00000004;
240 
241   public final static int CA_S_MSG_NO     = 0x0D;
242   public final static int CA_S_SEVERITY   = 0x03;
243 
244   public final static int CA_V_MSG_NO     = 0x03;
245   public final static int CA_V_SEVERITY   = 0x00;
246   public final static int CA_V_SUCCESS    = 0x00;
247 
248   /**
249    * Get EPICS status code.
250    * @return EPICS status code.
251    */
252   public final int getStatusCode()
253   {
254   	return ((getValue() << CA_V_MSG_NO) & CA_M_MSG_NO) | 
255 		   ((getSeverity().getValue() << CA_V_SEVERITY) & CA_M_SEVERITY);
256   }
257 
258   /**
259    * Get status instance from EPICS status code.
260    * @param value EPICS status code.
261    * @return status instance from EPICS status code.
262    */
263   static final public CAStatus forStatusCode(final int value)
264   {
265   	return forValue((value & CA_M_MSG_NO) >> CA_V_MSG_NO);
266   }
