26  package gov.aps.jca.event;
27  
28  import gov.aps.jca.*;
29  import gov.aps.jca.dbr.*;
30  
31  /**
32   * An event which indicates the completion of an asynchronous Get request.
33   *
34   * @see gov.aps.jca.Channel
35   */
36  public class GetEvent extends CAEvent {
37  
38    protected DBR _dbr;
39  
40    protected CAStatus _status;
41  
42    public GetEvent( Channel channel, DBR dbr, CAStatus status ) {
43      super( channel );
44      _dbr=dbr;
45      _status=status;
46    }
47  
48    /**
49     * Returns the DBR object containing the requested value.
50     */
51    public DBR getDBR() {
52      return _dbr;
53    }
54  
55    public CAStatus getStatus() {
56      return _status;
57    }
58  
59  }
