36    protected int _count;

39    protected DBR(Object value) {
40  	  setValue(value);
41    }
42  
43    protected void setValue(Object value)
44    {
45      _value=value;
46      _count=java.lang.reflect.Array.getLength( _value );
47    }

50      return DBR.TYPE;
51    }

54      return _count;
55    }

58      return _value;
59    }

61    public abstract DBR convert(DBRType convertType) throws CAStatusException;
62    
63    public DBR convert(DBRType convertType, Object params) throws CAStatusException {
64  	  // normally conversion does not need external parameters, so make non-params method default
65  	  return convert(convertType);
66    }
67  
68    public void printInfo() {
69      printInfo( System.out );
70    }
71  
72    public void printInfo( PrintStream out ) {
73  //    out.println( "CLASS: "+getClass().getName() );
74      out.println( "TYPE: "+getType().getName() );
75      out.println( "COUNT: "+getCount() );
76      out.print( "VALUE: " );
77      Object value=getValue();
78      if( value instanceof byte[] ) {
79        printValue( ( byte[] )value, out );
80      } else if( value instanceof short[] ) {
81        printValue( ( short[] )value, out );
82      } else if( value instanceof int[] ) {
83        printValue( ( int[] )value, out );
84      } else if( value instanceof float[] ) {
85        printValue( ( float[] )value, out );
86      } else if( value instanceof double[] ) {
87        printValue( ( double[] )value, out );
88      } else if( value instanceof String[] ) {
89        printValue( ( String[] )value, out );
90      }
91      out.println();
92    }

199   /**
200    * Helper method to print an array of byte as a comma separated list to the standard output stream.
201    *
202    * @param value the array to print.
203    */
204   static public void printValue( byte[] value ) {
205     printValue( value, System.out );
206   }
207 
208   /**
209    * Helper method to print an array of byte as a comma separated list to the specified output stream.
210    *
211    * @param value the array to print.
212    * @param out the output stream.
213    */
214   static public void printValue( byte[] value, PrintStream out ) {
215     for( int t=0; t<value.length; ++t ) {
216       if( t>0 ) {
217         out.print( "," );
218       }
219       out.print( value[t] );
220     }
221   }
