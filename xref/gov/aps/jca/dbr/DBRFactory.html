26  package gov.aps.jca.dbr;
27  
28  public class DBRFactory {
29  
30    static public DBR create( DBRType type, int count ) {
31      return type.newInstance( count );
32    }
33  
34    static public DBR create(int type, int count) {
35      DBRType t= DBRType.forValue(type);
36      if(t==null) return null;
37      return create(t,count);
38    }

40  
41  }
