15  package gov.aps.jca.cas;
16  
17  import gov.aps.jca.CAException;
18  import gov.aps.jca.CAStatusException;
19  
20  import java.net.InetSocketAddress;
21  
22  /**
23   * CA Server interface.
24   * User has to implement this interface to create <code>ServerContext</code>.
25   * ServerContext are created through the JCALibrary factory class.
26   *
27   * @author <a href="mailto:matej.sekoranjaATcosylab.com">Matej Sekoranja</a>
28   * @version $Id: Server.java,v 1.3 2006-08-22 15:11:38 msekoranja Exp $
29   * @see gov.aps.jca.JCALibrary
30   */
31  public interface Server {
32    
33    /**
34     * This function is called by the server library when it needs to
35     * determine if a named process variable (PV) exists (or could be created) in the
36     * server tool.
37     * The request is allowed to complete asynchronously.
38     * 
39     * The server tool is encouraged to accept multiple PV name aliases for the same PV here.
40     *
41     * NOTE: return <code>ProcessVariableExistanceCompletion.DOES_NOT_EXIST_HERE</code> if too many simultaneous
42     * asynchronous operations are pending against the server.
43     * The client library will retry the request at some time in the future.
44     * 
45     * @param aliasName the process variable alias name.
46     * @param clientAddress the client which requested test.
47     * @param	asyncCompletionCallback if asynchronous completion is required method should return 
48     * 								<code>ProcessVariableExistanceCompletion.ASYNC_COMPLETION</code> and call
49     * 								<code>ProcessVariableExistanceCompletionCallback.processVariableExistanceTestCompleted()</code> method.
50     * @return process variable existance status.
51     * @throws java.lang.IllegalArgumentException is thrown if the process variable name is invalid.
52     * @throws java.lang.IllegalStateException if the context has been destroyed.
53     */
54    public ProcessVariableExistanceCompletion processVariableExistanceTest(String aliasName, InetSocketAddress clientAddress, 
55  																		  ProcessVariableExistanceCallback asyncCompletionCallback)
56    	throws CAException, IllegalArgumentException, IllegalStateException;
57  
58    /**
59     * This function is called <b>every time</b> that a client attaches to the PV.
60     * The name supplied here will be either a canonical PV name or an alias PV name.
61     * The request is allowed to complete asynchronously.
62     * 
63     * It is a responsibility of the server tool to detect attempts by
64     * the server library to attach to an existing PV. If the PV does not
65     * exist then the server tool should create it. Otherwise, the server
66     * tool typically will return a pointer to the preexisting PV.
67     * 
68     * The server tool is encouraged to accept multiple PV name aliases
69     * for the same PV here.
70     *  
71     * In most situations the server tool should avoid PV duplication
72     * by returning a pointer to an existing PV if the PV alias name
73     * matches a preexisting PV's name or any of its aliases.
74     * In certain specialized rare situations the server tool may choose
75     * to create client private process variables that are not shared between
76     * clients. In this situation there might be several process variables
77     * with the same name. One for each client. For example, a server tool
78     * might be written that provides access to archival storage of the
79     * historic values of a set of process variables. Each client would
80     * specify its date of interest by writing into a client private process
81     * variable. Next the client would determine the current value of its
82     * subset of named process variables at its privately specified date by
83     * attaching to additional client private process variables.
84     *
85     * @param aliasName the process variable alias name.
86     * @param eventCallback the process variable event callback (where PV reports value changes if <code>interest</code> is <code>true</code>).
87     * @param	asyncCompletionCallback if asynchronous completion is required method should return 
88     * 								<code>null</code> and call
89     * 								<code>ProcessVariableExistanceCompletionCallback.processVariableAttachCompleted()</code> method.
90     * @return requested <code>ProcessVariable</code> instance, <code>null</code> if operation is to be done asynchronously or
91     * 		 exception is thrown in case of an error (non-existant PV, etc.).
92     *
93     * @throws CAStatusException is thrown if a Channel Access error occured while creating the channel.
94     * @throws java.lang.IllegalArgumentException is thrown if the channel's name is null or empty.
95     * @throws java.lang.IllegalStateException if the context has been destroyed.
96     */
97    public ProcessVariable processVariableAttach(String aliasName, ProcessVariableEventCallback eventCallback,
98  		  										ProcessVariableAttachCallback asyncCompletionCallback)
99    	throws CAStatusException, IllegalArgumentException, IllegalStateException;
100 
101 }
