15  package com.cosylab.epics.caj.impl;
16  
17  import java.io.IOException;
18  import java.net.InetSocketAddress;
19  
20  import com.cosylab.epics.caj.impl.CAContext;
21  
22  /**
23   * Interface defining transport (connection).
24   * @author <a href="mailto:matej.sekoranjaATcosylab.com">Matej Sekoranja</a>
25   * @version $id$
26   */
27  public interface Transport {
28  
29  	/**
30  	 * Get remote address.
31  	 * @return remote address.
32  	 */
33  	public InetSocketAddress getRemoteAddress();
34  
35  	/**
36  	 * Get context transport is living in.
37  	 * @return context transport is living in.
38  	 */
39  	public CAContext getContext();
40  	
41  	/**
42  	 * Transport protocol minor revision.
43  	 * @return protocol minor revision.
44  	 */
45  	public short getMinorRevision();
46  	
47  	/**
48  	 * Transport priority.
49  	 * @return protocol priority.
50  	 */
51  	public short getPriority();
52  
53  	/**
54  	 * Enqueue (if supported or if necessary send immediately) request message.
55  	 * Message is sent immediately if its priority equals <code>SEND_IMMEDIATELY_PRIORITY</code>.
56  	 * @param requestMessage	request message to enqueue.
57  	 */
58  	public void submit(Request requestMessage) throws IOException;
59  	
60  	/**
61  	 * Enqueue (if necessary submit) request message.
62  	 * @return <code>true</code> on success, <code>false</code> on failure.
63  	 */
64  	public boolean flush();
65  
66  }
