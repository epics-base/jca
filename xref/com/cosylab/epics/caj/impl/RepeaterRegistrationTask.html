15  package com.cosylab.epics.caj.impl;
16  
17  import java.io.IOException;
18  import java.net.InetSocketAddress;
19  import java.net.UnknownHostException;
20  import java.util.logging.Level;
21  
22  import com.cosylab.epics.caj.CAJContext;
23  import com.cosylab.epics.caj.impl.requests.RepeaterRegisterRequest;
24  import com.cosylab.epics.caj.util.Timer;
25  
26  /**
27   * Repeater registration tasks.
28   * Periodically issues reppeater registration requests until registration is not confirmed.
29   * @author <a href="mailto:matej.sekoranjaATcosylab.com">Matej Sekoranja</a>
30   * @version $id$
31   */
32  public class RepeaterRegistrationTask implements Timer.TimerRunnable {
33  
34  	private CAJContext context;
35  	private RepeaterRegisterRequest registrationRequest; 
36  	private InetSocketAddress repeaterAddress;
37  	private Object timerID;
38  	
39  	/**
40  	 * @param context
41  	 * @param repeaterAddress
42  	 * @throws UnknownHostException
43  	 */
44  	public RepeaterRegistrationTask(CAJContext context, 
45  									InetSocketAddress repeaterAddress) throws UnknownHostException
46  	{
47  		this.context = context;
48  		this.repeaterAddress = repeaterAddress;
49  		 
50  		this.registrationRequest = new RepeaterRegisterRequest(context.getBroadcastTransport());
51  	}
52  
53  	/**
54  	 * @see com.cosylab.epics.caj.util.Timer.TimerRunnable#timeout(long)
55  	 */
56  	public void timeout(long timeToRun) {
57  		// repeater confirmed registration, cancel timer
58  		if (context.isRegistrationConfirmed())
59  			Timer.cancel(timerID);
60  		else
61  			registrationRequest();
62  	}
63  
64  	/**
65  	 * Schedules registration requests every <code>period</code> ms.
66  	 * @param period	period in ms.
67  	 */
68  	public void runInBackground(int period)
69  	{
70  		timerID = context.getTimer().executePeriodically(period, this, 0);
71  	}
72  
73  	/**
74  	 * Issue registration request to repeater.
75  	 */
76  	public void registrationRequest()
77  	{
78  		context.getLogger().log(Level.FINE, "Repeater registation request sent to " + repeaterAddress + ".");
79  		try
80  		{
81  			context.getBroadcastTransport().send(registrationRequest, repeaterAddress);
82  		}
83  		catch (IOException ioex)
84  		{
85  			context.getLogger().log(Level.FINE, "Failed to send registation request.", ioex);
86  		}
87  	}
88  
89  }
