15  package com.cosylab.epics.caj.impl;
16  
17  import java.net.InetSocketAddress;
18  
19  /**
20   * Connection exception.
21   * @author <a href="mailto:matej.sekoranjaATcosylab.com">Matej Sekoranja</a>
22   * @version $id$
23   */
24  public class ConnectionException extends Exception {
25  
26  	static final long serialVersionUID = 2533297215333289327L;
27  
28  	/**
29  	 * Failed connection address.
30  	 */
31  	private InetSocketAddress address;
32  
33  	/**
34  	 * @param message
35  	 * @param cause
36  	 */
37  	public ConnectionException(String message, InetSocketAddress address, Throwable cause) {
38  		super(message, cause);
39  		this.address = address;
40  	}
41  
42      /**
43       * Get connection addresss.
44       * @return connection address.
45       */
46      public InetSocketAddress getAddress() {
47          return address;
48      }
49  }
