15  package com.cosylab.epics.caj.impl.sync;
16  
17  import java.util.concurrent.TimeUnit;
18  import java.util.concurrent.atomic.AtomicInteger;
19  import java.util.concurrent.locks.Lock;
20  import java.util.concurrent.locks.ReentrantLock;
21  
22  /**
23   * Reference couting mutex implementation w/ deadlock detection.
24   * Synchronization helper class used (intended for use) for activation/deactivation synchronization.
25   * This class enforces <code>attempt</code> method of acquiring the locks to prevent deadlocks.
26   * Class also offers reference counting.
27   * (NOTE: automatic lock counting was not implemented due to imperfect usage.)
28   * 
29   * Example of usage:
30   * <code>
31   *		ReferenceCountingLock lock;
32   * 		if (lock.acquire(3*Sync.ONE_MINUTE))
33   * 		{
34   * 			try
35   * 			{
36   * 				// critical section here
37   * 			}
38   * 			finally
39   * 			{
40   * 				lock.release();
41   * 			}
42   * 		}	
43   * 		else
44   * 		{
45   * 			throw new TimoutException("Deadlock detected...");
46   * 		}
47   * 		
48   * </code>
49   * @author <a href="mailto:matej.sekoranjaATcosylab.com">Matej Sekoranja</a>
50   * @version $id$
51   */
52  class ReferenceCountingLock
53  {
54  	/**
55  	 * Number of current locks.
56  	 */
57  	private AtomicInteger references = new AtomicInteger(1);
58  
59  	/**
60  	 * Synchronization mutex.
61  	 */
62  	private Lock lock = new ReentrantLock();
63  		
64  	/**
65  	 * Constructor of <code>ReferenceCountingLock</code>.
66  	 * After construction lock is free and reference count equals <code>1</code>.
67  	 */
68  	public ReferenceCountingLock()
69  	{
70  		// no-op.
71  	}
72  		
73  	/**
74  	 * Attempt to acquire lock.
75  	 * 
76  	 * @param	msecs	the number of milleseconds to wait.
77  	 * 					An argument less than or equal to zero means not to wait at all.
78  	 * @return	<code>true</code> if acquired, <code>false</code> othwerwise.
79  	 */
80  	public boolean acquire(long msecs)
81  	{
82  		try
83  		{
84  			return lock.tryLock(msecs, TimeUnit.MILLISECONDS);
85  		}
86  		catch (InterruptedException ie)
87  		{
88  			return false;
89  		}
90  	}
91  
92  	/**
93  	 * Release previously acquired lock.
94  	 */
95  	public void release()
96  	{
97  		lock.unlock();
98  	}
99  		
100 	/*
101 	 * Get number of references.
102 	 * 
103 	 * @return number of references.
104 	 *
105 	public int referenceCount()
106 	{
107 		return references.get();
108 	}*/
109 
110 	/**
111 	 * Increment number of references.
112 	 * 
113 	 * @return number of references.
114 	 */
115 	public int increment()
116 	{
117 		return references.incrementAndGet();
118 	}
119 
120 	/**
121 	 * Decrement number of references.
122 	 * 
123 	 * @return number of references.
124 	 */
125 	public int decrement()
126 	{
127 		return references.decrementAndGet();
128 	}
129 
130 }
