15  package com.cosylab.epics.caj.impl.sync;
16  
17  import java.util.HashMap;
18  import java.util.Map;
19  
20  /**
21   * Named lock implementation (I named it "Named-lock pattern").
22   * <code>
23   * // try to acquire named lock
24   * boolean lockAcquired = namedLocker.acquireSynchronizationObject(namedObject, getLockTimeout());
25   * if (lockAcquired) 
26   * {
27   * 		try
28   *		{
29   *		    // ... so sth here
30   *		}
31   *		finally
32   *		{
33   *			namedLocker.releaseSynchronizationObject(name);	
34   *		}
35   *	}
36   *	else
37   *	{
38   *		// .. failed to obtain synchronization lock for component 'namedObject', possible deadlock
39   *	}
40   * </code>
41   * @author <a href="mailto:matej.sekoranjaATcosylab.com">Matej Sekoranja</a>
42   * @version $id$
43   */
44  public class NamedLockPattern {
45  
46  	/**
47  	 * Map of (named) locks.
48  	 */
49  	final private Map<Object, ReferenceCountingLock> namedLocks;
50  
51  	/**
52  	 * Default constructor. 
53  	 */
54  	public NamedLockPattern() {
55  		namedLocks = new HashMap<>();
56  	}
57  
58  	/**
59  	 * Acquire synchronization lock for named object.
60  	 * @param	name	name of the object whose lock to acquire.
61  	 * @param	msec	the number of milleseconds to wait.
62  	 * 					An argument less than or equal to zero means not to wait at all.
63  	 * @return	<code>true</code> if acquired, <code>false</code> othwerwise.
64  	 */
65  	public boolean acquireSynchronizationObject(Object name, long msec)
66  	{
67  		ReferenceCountingLock lock;
68  		
69  		synchronized (namedLocks)
70  		{
71  			// get synchronization object
72  			lock = namedLocks.get(name);
73  		
74  			// none is found, create and return new one
75  			// increment references
76  			if (lock == null)
77  			{
78  				lock = new ReferenceCountingLock();
79  				namedLocks.put(name, lock);
80  			}
81  			else
82  				lock.increment();
83  		}
84  		
85  		boolean success = lock.acquire(msec);
86  		
87  		if (!success)
88  			releaseSynchronizationObject(name, false);
89  		
90  		return success;
91  	}
92  
93  	/**
94  	 * Release synchronization lock for named object.
95  	 * @param	name	name of the object whose lock to release.
96  	 */
97  	public void releaseSynchronizationObject(Object name)
98  	{
99  		releaseSynchronizationObject(name, true);
100 	}
101 
102 	/**
103 	 * Release synchronization lock for named object.
104 	 * @param	name	name of the object whose lock to release.
105 	 * @param	release	set to <code>false</code> if there is no need to call release
106 	 * 					on synchronization lock.
107 	 */
108 	public void releaseSynchronizationObject(Object name, boolean release)
109 	{
110 		synchronized (namedLocks)
111 		{
112 			// get synchronization object
113 			ReferenceCountingLock lock = namedLocks.get(name);
114 		
115 			// release lock
116 			if (lock != null)
117 			{
118 				// if there only one current lock exists
119 				// remove it from the map
120 				if (lock.decrement() <= 0)
121 					namedLocks.remove(name);
122 					
123 				// release the lock
124 				if (release)
125 					lock.release();
126 			}
127 		}
128 	}
129 
130 }
