15  package com.cosylab.epics.caj.impl;
16  
17  /**
18   * CA constants.
19   * @author <a href="mailto:matej.sekoranjaATcosylab.com">Matej Sekoranja</a>
20   * @version $id$
21   */
22  public interface CAConstants {
23  
24  	/**
25  	 * CA protocol major revision (implemented by this library).
26  	 */
27  	public static final short CA_MAJOR_PROTOCOL_REVISION = 4;
28  	
29  	/**
30  	 * CAC (client) protocol minor revision (implemented by this library).
31  	 */
32  	public static final short CA_MINOR_PROTOCOL_REVISION = 13;
33  
34  	/**
35  	 * CAS (server) protocol minor revision (implemented by this library).
36  	 */
37  	public static final short CAS_MINOR_PROTOCOL_REVISION = 11;
38  
39  	/**
40  	 * Unknown CA protocol minor revision.
41  	 */
42  	public static final short CA_UNKNOWN_MINOR_PROTOCOL_REVISION = 0;
43  
44  	/**
45  	 * CA protocol port base.
46  	 */
47  	public static final int CA_PORT_BASE = 5056;
48  
49  	/**
50  	 * Default CA server port.
51  	 */
52  	public static final int CA_SERVER_PORT = CA_PORT_BASE + 2 * CA_MAJOR_PROTOCOL_REVISION;
53  
54  	/**
55  	 * Default CA repeater port.
56  	 */
57  	public static final int CA_REPEATER_PORT = CA_PORT_BASE + 2 * CA_MAJOR_PROTOCOL_REVISION + 1;
58  
59  	/**
60  	 * CA protocol message header size.
61  	 */
62  	public static final short CA_MESSAGE_HEADER_SIZE = 16;
63  	
64  	/**
65  	 * CA protocol message extedned header size.
66  	 */
67  	public static final short CA_EXTENDED_MESSAGE_HEADER_SIZE = CA_MESSAGE_HEADER_SIZE + 8;
68  
69  	/**
70  	 * UDP maximum send message size.
71  	 * MAX_UDP: 1500 (max of ethernet and 802.{2,3} MTU) - 20(IP) - 8(UDP) 
72  	 * (the MTU of Ethernet is currently independent of its speed varient)
73  	 */
74  	public static final int MAX_UDP_SEND = 1024;
75  
76  	/**
77  	 * UDP maximum receive message size.
78  	 */
79  	public static final int MAX_UDP_RECV = 0xFFFF + 16;
80  
81  	/**
82  	 * TCP maximum receive message size.
83  	 */
84  	public static final int MAX_TCP_RECV = 1024 * 16 + CA_EXTENDED_MESSAGE_HEADER_SIZE;
85  
86  	/**
87  	 * Default priority (corresponds to POSIX SCHED_OTHER)
88  	 */
89  	public static final short CA_DEFAULT_PRIORITY = 0;
90  	
91  	/**
92  	 * Read access right mask.
93  	 */
94  	public static final int CA_PROTO_ACCESS_RIGHT_READ = 1 << 0;
95  
96  	/**
97  	 * Write access right mask.
98  	 */
99  	public static final int CA_PROTO_ACCESS_RIGHT_WRITE = 1 << 1;
100 
101 	/**
102 	 * Do not require response for CA search request.
103 	 */
104 	public static final short CA_SEARCH_DONTREPLY = 5;
105 
106 	/**
107 	 * Require response (even if not found) for CA search request over TCP.
108 	 */
109 	public static final short CA_SEARCH_DOREPLY = 10;
110 
111 	/**
112 	 * Echo (state-of-health message) reponse timeput in ms.
113 	 */
114 	public static final long CA_ECHO_TIMEOUT = 5000;
115 
116 	/**
117      * Max. (requested) string size.
118      */
119 	public static final int MAX_STRING_SIZE = 40;
120 	
121 	/**
122 	 * Unreasonable channel name length.
123 	 */
124 	public static final int UNREASONABLE_CHANNEL_NAME_LENGTH = 500;
125 }
