39  /**
40   * CA response handler - main handler which dispatches responses to appripriate handlers.
41   * @author <a href="mailto:matej.sekoranjaATcosylab.com">Matej Sekoranja</a>
42   * @version $id$
43   */
44  public class CAResponseHandler implements ResponseHandler {
45  
46  	/**
47  	 * Table of response handlers for each command ID.
48  	 */
49  	private ResponseHandler[] handlerTable;
50  
51  	/**
52  	 * Context instance.
53  	 */
54  	private CAJContext context;
55  	
56  	/**
57  	 * @param context
58  	 */
59  	public CAResponseHandler(CAJContext context) {
60  		this.context = context;
61  		intializeHandlerTable();
62  	}
63  
64  	/**
65  	 * Initialize handler table.
66  	 */
67  	private void intializeHandlerTable()
68  	{
69  		ResponseHandler badResponse = new BadResponse(context);
70  		
71  		handlerTable = new ResponseHandler[]
72  			{
73  				new VersionResponse(context), /*  0 */
74  				new EventAddResponse(context), /*  1 */
75  				new EventCancelResponse(context), /*  2 */
76  				badResponse, /*  3 */
77  				badResponse, /*  4 */
78  				badResponse, /*  5 */
79  				new SearchResponse(context), /*  6 */
80  				badResponse, /*  7 */
81  				badResponse, /*  8 */
82  				badResponse, /*  9 */
83  				badResponse, /* 10 */
84  				new ExceptionResponse(context), /* 11 */
85  				new ClearChannelResponse(context), /* 12 */
86  				new BeaconResponse(context), /* 13 */
87  				new NoopResponse(context, "Channel not found"), /* 14 */
88  				new ReadNotifyResponse(context), /* 15 */
89  				badResponse, /* 16 */
90  				new RepeaterConfirmResponse(context), /* 17 */
91  				new CreateChannelResponse(context), /* 18 */
92  				new WriteNotifyResponse(context), /* 19 */
93  				badResponse, /* 20 */
94  				badResponse, /* 21 */
95  				new AccessRightsResponse(context), /* 22 */
96  				new EchoResponse(context), /* 23 */
97  				badResponse, /* 24 */
98  				badResponse, /* 25 */
99  				new CreateChannelFailedResponse(context), /* 26 */
100 				new ChannelDisconnectedResponse(context), /* 27 */
101 			};
102 	}
103 	
104 	/**
105 	 * @see com.cosylab.epics.caj.impl.ResponseHandler#handleResponse(java.net.InetSocketAddress, com.cosylab.epics.caj.impl.Transport, java.nio.ByteBuffer[])
106 	 */
107 	public void handleResponse(
108 		InetSocketAddress responseFrom,
109 		Transport transport,
110 		ByteBuffer[] response) {
111 			
112 		ByteBuffer headerBuffer = response[0];
113 		int headerBufferSize = headerBuffer.limit() - headerBuffer.position();
114 		if (headerBufferSize < CAConstants.CA_MESSAGE_HEADER_SIZE)
115 		{
116 			context.getLogger().fine("Invalid response header (" + CAConstants.CA_MESSAGE_HEADER_SIZE + " bytes expected, " + headerBufferSize + " received) from " + responseFrom + ".");
117 			// TODO remove debug output
118 			// XXX switching to direct buffers, headerBuffer.array() may not be available
119 			// HexDump.hexDump("Invalid CA Header", headerBuffer.array(), headerBuffer.position(), headerBufferSize);
120 			// flush all header buffer (needed for UDP packed responses)
121 			headerBuffer.position(headerBuffer.limit());
122 			return;
123 		}
124 
125 		// peek for reponse ID
126 		short commandID = headerBuffer.getShort(headerBuffer.position());
127 		if (commandID < 0 || commandID >= handlerTable.length)
128 		{
129 			context.getLogger().fine("Invalid (or unsupported) command: " + commandID + ".");
130 			// TODO remove debug output
131 			// XXX switching to direct buffers, headerBuffer.array() may not be available
132 			// HexDump.hexDump("Invalid CA Header", headerBuffer.array(), headerBuffer.position(), headerBufferSize);
133 			// flush all header buffer (needed for UDP packed responses)
134 			headerBuffer.position(headerBuffer.limit());
135 			return;
136 		}
137 		
138 		//delegate
139 		handlerTable[commandID].handleResponse(responseFrom, transport, response);
140 
141 	}
