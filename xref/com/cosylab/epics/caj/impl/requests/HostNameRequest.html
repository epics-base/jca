15  package com.cosylab.epics.caj.impl.requests;
16  
17  import java.net.InetAddress;
18  import java.nio.ByteBuffer;
19  
20  import com.cosylab.epics.caj.CAJConstants;
21  import com.cosylab.epics.caj.impl.CAConstants;
22  import com.cosylab.epics.caj.impl.Request;
23  import com.cosylab.epics.caj.impl.Transport;
24  
25  /**
26   * @author <a href="mailto:matej.sekoranjaATcosylab.com">Matej Sekoranja</a>
27   * @version $id$
28   */
29  public class HostNameRequest extends AbstractCARequest {
30  
31  	private static String hostName = null;
32  	private static final String HOSTNAME_KEY = "HOSTNAME";
33  	
34  	private static synchronized String getHostName() 
35  	{
36  		if (hostName == null)
37  		{
38  			// default fallback
39  			hostName = "localhost";
40  			
41  			try {
42  				InetAddress localAddress = InetAddress.getLocalHost();
43  				hostName = localAddress.getHostName();
44  			} catch (Throwable uhe) {	// not only UnknownHostException
45  				// try with environment variable
46  				try {
47  					String envHN = System.getenv(HOSTNAME_KEY);
48  					if (envHN != null)
49  						hostName = envHN;
50  				} catch (Throwable th) {
51  					// in case not supported by JVM/OS
52  				}
53  				
54  				// and system property (overrides env. var.)
55  				hostName = System.getProperty(HOSTNAME_KEY, hostName);
56  			}
57  			
58  			if (System.getProperties().contains(CAJConstants.CAJ_STRIP_HOSTNAME))
59  			{
60  				int dotPos = hostName.indexOf('.');
61  				if (dotPos > 0)
62  					hostName = hostName.substring(0, dotPos);
63  			}
64  		}
65  		
66  		return hostName;
67  	}
68  	
69  	/**
70  	 * @param transport transport to be used when sending.
71  	 */
72  	public HostNameRequest(Transport transport) {
73  		super(transport);
74  		
75  		// compatibility check
76  		if (transport.getMinorRevision() < 1)
77  			return;
78  
79  		String hostName = getHostName();
80  		
81  		int alignedMessageSize = calculateAlignedSize(8, CAConstants.CA_MESSAGE_HEADER_SIZE + hostName.length() + 1);
82  		requestMessage = ByteBuffer.allocate(alignedMessageSize);
83  		requestMessage = insertCAHeader(transport, requestMessage,
84  										(short)21, (short)(alignedMessageSize - CAConstants.CA_MESSAGE_HEADER_SIZE),
85  										(short)0, (short)0,
86  										0, 0);
87  		// append zero-terminated string and align message
88  		requestMessage.put(hostName.getBytes());
89  		requestMessage.put((byte)0);
90  		requestMessage = alignBuffer(8, requestMessage);
91  	}
92  
93  	/**
94  	 * @see com.cosylab.epics.caj.impl.Request#getPriority()
95  	 */
96  	public byte getPriority() {
97  		return Request.SEND_IMMEDIATELY_PRIORITY;
98  	}
99  
100 }
