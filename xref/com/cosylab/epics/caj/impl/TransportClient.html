15  package com.cosylab.epics.caj.impl;
16  
17  /**
18   * Client (user) of the transport.
19   * @author <a href="mailto:matej.sekoranjaATcosylab.com">Matej Sekoranja</a>
20   * @version $id$
21   */
22  public interface TransportClient {
23  	
24  	/**
25  	 * Notification of unresponsive transport (e.g. no heartbeat detected) .
26  	 */
27  	public void transportUnresponsive();
28  
29  	/**
30  	 * Notification of responsive transport (e.g. heartbeat detected again),
31  	 * called to discard <code>transportUnresponsive</code> notification.
32  	 * @param transport	responsive transport.
33  	 */
34  	public void transportResponsive(Transport transport);
35  
36  	/**
37  	 * Notification of network change (server restarted).
38  	 */
39  	public void transportChanged();
40  
41  	/**
42  	 * Notification of forcefully closed transport.
43  	 */
44  	public void transportClosed();
45  
46  }
