15  package com.cosylab.epics.caj.impl.handlers;
16  
17  import java.nio.ByteBuffer;
18  
19  import com.cosylab.epics.caj.CAJContext;
20  import com.cosylab.epics.caj.impl.ExceptionHandler;
21  import com.cosylab.epics.caj.impl.ResponseRequest;
22  
23  /**
24   * @author <a href="mailto:matej.sekoranjaATcosylab.com">Matej Sekoranja</a>
25   * @version $id$
26   */
27  public class ResponseRequestExceptionHandler implements ExceptionHandler {
28  
29  	/**
30  	 * Context instance.
31  	 */
32  	protected CAJContext context;
33  
34  	/**
35  	 * @param context
36  	 */
37  	public ResponseRequestExceptionHandler(CAJContext context) {
38  		this.context = context;
39  	}
40  
41  	/**
42  	 * @see com.cosylab.epics.caj.impl.ExceptionHandler#handleException(int, int, java.lang.String, java.nio.ByteBuffer)
43  	 */
44  	public void handleException(
45  		int errorCode,
46  		int cid,
47  		String errorMessage,
48  		ByteBuffer originalHeaderBuffer) {
49  
50  		int ioid = originalHeaderBuffer.getInt(12);
51  		ResponseRequest rr = (ResponseRequest)context.getResponseRequest(ioid);
52  		if (rr == null)
53  			return;
54  			
55  		rr.exception(errorCode, errorMessage);					
56  
57  	}
58  
59  }
