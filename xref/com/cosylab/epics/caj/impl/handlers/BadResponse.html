15  package com.cosylab.epics.caj.impl.handlers;
16  
17  import java.net.InetSocketAddress;
18  import java.nio.ByteBuffer;
19  
20  import com.cosylab.epics.caj.CAJContext;
21  import com.cosylab.epics.caj.impl.Transport;
22  import com.cosylab.epics.caj.util.HexDump;
23  
24  /**
25   * Bad response handler.
26   * @author <a href="mailto:matej.sekoranjaATcosylab.com">Matej Sekoranja</a>
27   * @version $id$
28   */
29  public class BadResponse extends AbstractCAJResponseHandler {
30  
31  	/**
32  	 * @param context
33  	 */
34  	public BadResponse(CAJContext context) {
35  		super(context, "Bad response");
36  	}
37  
38  	/**
39  	 * @see com.cosylab.epics.caj.impl.handlers.AbstractCAJResponseHandler#internalHandleResponse(java.net.InetSocketAddress, com.cosylab.epics.caj.impl.Transport, java.nio.ByteBuffer[])
40  	 */
41  	protected void internalHandleResponse(
42  		InetSocketAddress responseFrom,
43  		Transport transport,
44  		ByteBuffer[] response) {
45  		// noop
46  	}
47  
48  	/**
49  	 * @see com.cosylab.epics.caj.impl.ResponseHandler#handleResponse(java.net.InetSocketAddress, com.cosylab.epics.caj.impl.Transport, java.nio.ByteBuffer[])
50  	 */
51  	public void handleResponse(
52  		InetSocketAddress responseFrom,
53  		Transport transport,
54  		ByteBuffer[] response)
55  	{
56  			ByteBuffer headerBuffer = response[0];
57  			int startPos = headerBuffer.position();
58  			int command = headerBuffer.getShort();
59  			
60  			context.getLogger().fine("Undecipherable message (bad response type " + command + ") from " + responseFrom + ".");
61  
62  			// TODO remove debug output
63  			HexDump.hexDump(description, headerBuffer.array(), startPos, headerBuffer.limit() - startPos);
64  			
65  			// flush all header buffer (needed for UDP packed responses)
66  			headerBuffer.position(headerBuffer.limit());
67  	}
68  
69  }
