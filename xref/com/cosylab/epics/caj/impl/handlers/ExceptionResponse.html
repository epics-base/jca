98  	/**
99  	 * @see com.cosylab.epics.caj.impl.handlers.AbstractCAJResponseHandler#internalHandleResponse(java.net.InetSocketAddress, com.cosylab.epics.caj.impl.Transport, java.nio.ByteBuffer[])
100 	 */
101 	protected void internalHandleResponse(
102 		InetSocketAddress responseFrom,
103 		Transport transport,
104 		ByteBuffer[] response) {
105 			
106 		ByteBuffer originalHeaderBuffer = null;
107 		String errorMessage = null;
108 
109 		ByteBuffer payloadBuffer;
110 		if (response.length == 1)
111 			// UDP response
112 			payloadBuffer = response[0];
113 		else
114 			// TCP response
115 			payloadBuffer = response[1];
116 			
117 		int payloadStart = payloadBuffer.position();
118 			
119 		// zero termination string expected, i.e. at least CAConstants.CA_MESSAGE_HEADER_SIZE + 1 bytes
120 		if (payloadSize > CAConstants.CA_MESSAGE_HEADER_SIZE)
121 		{
122 			int originalHeaderPayloadSize = payloadBuffer.getShort(payloadStart + 2) & 0xFFFF;
123 					
124 			// extended message header check
125 			int originalHeaderSize;
126 			if (originalHeaderPayloadSize == 0xFFFF)
127 				originalHeaderSize = CAConstants.CA_EXTENDED_MESSAGE_HEADER_SIZE;
128 			else
129 				originalHeaderSize = CAConstants.CA_MESSAGE_HEADER_SIZE;
130 			
131 			originalHeaderBuffer = payloadBuffer.slice();
132                         originalHeaderBuffer.limit(originalHeaderSize);
133 
134 			// find zero-termination (from end is efficient, but not error-proof)
135 			int errorMessageStart = payloadStart + originalHeaderSize;
136 			int errorMessageEnd = errorMessageStart;
137 			while (payloadBuffer.get(errorMessageEnd) != 0)
138 				errorMessageEnd++;
139                         payloadBuffer.position(errorMessageStart);
140                         ByteBuffer errorMessageBuffer = payloadBuffer.slice();
141                         errorMessageBuffer.limit(errorMessageEnd - errorMessageStart);
142                         errorMessage = Charset.defaultCharset().decode(errorMessageBuffer).toString();
143 		}
144 		
145 		// read rest of the payload (needed for UDP)
146 		payloadBuffer.position(payloadStart + payloadSize);
147 		
148 		//
149 		// delegate
150 		//
151 		
152 		// peek for command ID from original header
153 		short commandID = (originalHeaderBuffer != null ? originalHeaderBuffer.getShort(0) : -1);
154 		if (commandID < 0 || commandID >= handlerTable.length)
155 		{
156 			context.getLogger().fine("Invalid (or unsupported) exception command: " + commandID + ".");
157 			this.handleException(parameter2, parameter1, errorMessage, originalHeaderBuffer);
158 			return;
159 		}
160 		
161 		// TODO remove
162 		if (debug)
163 			context.getLogger().fine("Exception occured, code: " + CAStatus.forStatusCode(parameter2) + ", message: '" + errorMessage + "'.");
164 System.err.println("Exception occured, code: " + CAStatus.forStatusCode(parameter2) + ", message: '" + errorMessage + "'.");	
165 		//delegate
166 		/* exception code, channel id, error message, original header) */
167 		handlerTable[commandID].handleException(parameter2, parameter1, errorMessage, originalHeaderBuffer);
168 		
169 	}
