15  package com.cosylab.epics.caj.impl.reactor.lf;
16  
17  import java.nio.channels.SelectionKey;
18  
19  import com.cosylab.epics.caj.impl.reactor.Reactor;
20  import com.cosylab.epics.caj.impl.reactor.ReactorHandler;
21  
22  /**
23   * Decorator pattern to 'decorate' application-specific event processing handler
24   * to work with Leader/Followers deisgn pattern. This decorator promotes a new leader
25   * thread and disables/enables the handler. 
26   * @author <a href="mailto:matej.sekoranjaATcosylab.com">Matej Sekoranja</a>
27   * @version $id$
28   */
29  public class LeaderFollowersHandler implements ReactorHandler, Runnable {
30      
31      /**
32       * Reactor to serve.
33       */
34      protected Reactor reactor;
35  
36      /**
37       * Application-specific event processing handler.
38       */
39      protected ReactorHandler handler;
40      
41      /**
42       * Leader-followers thread pool.
43       */
44      protected LeaderFollowersThreadPool threadPool;
45  
46      /**
47       * Constructor.
48       * @param reactor reactor to handle.
49       * @param handler application-specific event processing handler.
50       * @param threadPool leader-followers thread pool.
51       */
52      public LeaderFollowersHandler(Reactor reactor, ReactorHandler handler, LeaderFollowersThreadPool threadPool) {
53          this.reactor = reactor;
54          this.handler = handler;
55          this.threadPool = threadPool;
56      }
57      
58      /**
59       * @see com.cosylab.epics.caj.impl.reactor.ReactorHandler#handleEvent(java.nio.channels.SelectionKey)
60       */
61      public void handleEvent(SelectionKey key)
62      {
63      	// if not valid, just promote new leader
64      	if (!key.isValid())
65      	{
66  	        threadPool.promoteLeader(this);
67  	        return;
68      	}
69      	
70          reactor.disableSelectionKey(key);
71          try
72          {
73  	        // promote a follower thread to become a leader
74  	        threadPool.promoteLeader(this);
75  //System.err.println("[processing] " + Thread.currentThread().getName());
76  	        // dispatch application-specific event processing code
77  	        handler.handleEvent(key);
78          }
79          finally
80          {
81              // enable the handle in the reactor
82              reactor.enableSelectionKey(key);
83          }
84      }
85  
86      /**
87       * @see java.lang.Runnable#run()
88       */
89      public void run()
90      {
91      	reactor.process();
92      }
93      
94  }
