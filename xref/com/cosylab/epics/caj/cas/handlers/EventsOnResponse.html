15  package com.cosylab.epics.caj.cas.handlers;
16  
17  import java.net.InetSocketAddress;
18  import java.nio.ByteBuffer;
19  
20  import com.cosylab.epics.caj.cas.CAJServerContext;
21  import com.cosylab.epics.caj.cas.CASTransport;
22  import com.cosylab.epics.caj.impl.Transport;
23  
24  /**
25   * Events on request handler.
26   * @author <a href="mailto:matej.sekoranjaATcosylab.com">Matej Sekoranja</a>
27   * @version $id$
28   */
29  public class EventsOnResponse extends AbstractCASResponseHandler {
30  
31  	/**
32  	 * @param context
33  	 */
34  	public EventsOnResponse(CAJServerContext context) {
35  		super(context, "Events on request");
36  	}
37  
38  	/**
39  	 * @see com.cosylab.epics.caj.impl.handlers.AbstractCAJResponseHandler#internalHandleResponse(java.net.InetSocketAddress, com.cosylab.epics.caj.impl.Transport, java.nio.ByteBuffer[])
40  	 */
41  	protected void internalHandleResponse(
42  		InetSocketAddress responseFrom,
43  		Transport transport,
44  		ByteBuffer[] response) {
45  		
46  		// turn events on
47  		((CASTransport)transport).eventsOn();
48  	}
49  
50  }
