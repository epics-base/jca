15  package com.cosylab.epics.caj.cas.handlers;
16  
17  import gov.aps.jca.CAStatus;
18  import gov.aps.jca.cas.ServerChannel;
19  import gov.aps.jca.cas.ServerMonitor;
20  
21  import java.net.InetSocketAddress;
22  import java.nio.ByteBuffer;
23  import java.util.logging.Level;
24  
25  import com.cosylab.epics.caj.cas.CAJServerContext;
26  import com.cosylab.epics.caj.cas.CASTransport;
27  import com.cosylab.epics.caj.cas.requests.EventCancelRequest;
28  import com.cosylab.epics.caj.impl.Transport;
29  
30  /**
31   * CA event cancel request handler.
32   * @author <a href="mailto:matej.sekoranjaATcosylab.com">Matej Sekoranja</a>
33   * @version $id$
34   */
35  public class EventCancelResponse extends AbstractCASResponseHandler {
36  
37  	/**
38  	 * @param context
39  	 */
40  	public EventCancelResponse(CAJServerContext context) {
41  		super(context, "Event cancel request");
42  	}
43  
44  	/**
45  	 * @see com.cosylab.epics.caj.impl.handlers.AbstractCAJResponseHandler#internalHandleResponse(java.net.InetSocketAddress, com.cosylab.epics.caj.impl.Transport, java.nio.ByteBuffer[])
46  	 */
47  	protected void internalHandleResponse(
48  		InetSocketAddress responseFrom,
49  		Transport transport,
50  		ByteBuffer[] response) {
51  
52  		CASTransport casTransport = (CASTransport)transport;
53  		
54  		// get channel by SID
55  		ServerChannel channel = casTransport.getChannel(parameter1);
56  		if (channel == null) 
57  		{
58  			sendException(transport, parameter1, CAStatus.BADCHID, response[0], null);
59  			disconnect(transport);
60  			return;
61  		} 
62  
63  		// get monitor by IOID
64  		ServerMonitor monitor = channel.getMonitor(parameter2);
65  		if (monitor == null) 
66  		{
67  			sendException(transport, parameter1, CAStatus.BADMONID, response[0], null);
68  			disconnect(transport);
69  			return;
70  		} 
71  		
72  		// destroy
73  		monitor.destroy();
74  		
75  		// send response back
76  		try
77  		{
78  			new EventCancelRequest(transport, parameter1, parameter2, dataType, dataCount).submit();
79  
80  		} catch (Throwable th) {
81  			context.getLogger().log(Level.WARNING, "Exception caught when responding to event cancel request for channel with SID : " + parameter1, th);
82  		}
83  		
84  	}
85  
86  	/**
87  	 * Disconnect.
88  	 */
89  	private void disconnect(Transport transport)
90  	{
91  		((CASTransport)transport).close(true);
92  	}
93  	
94  }
