42  public abstract class AbstractCASResponseHandler extends AbstractCAResponseHandler {
43  
44  	/**
45  	 * Context instance.
46  	 */
47  	protected CAJServerContext context;
48  
49  	/**
50  	 * @param context
51  	 * @param description
52  	 */
53  	public AbstractCASResponseHandler(CAJServerContext context, String description) {
54  		super(description);
55  		this.context = context;
56  	}
57  
58  	/**
59  	 * Extract string from payload.
60  	 * @param buffer buffer to use.
61  	 * @param start buffer offset to use, if negative current buffer position will be used
62  	 * @param maxSize maximum payload size reserved for string.
63  	 * @param setToEnd set to the end of string flag (after zero char)
64  	 * @return extracted string.
65  	 */
66  	public static final String extractString(ByteBuffer buffer, int start, int maxSize, boolean setToEnd)
67  	{
68  		if (start < 0)
69  			start = buffer.position();
70  		final int bufferEnd = start + maxSize;
71  
72  		byte[] data = buffer.array();
73  		int end = start;
74  		
75  		// find zero char (string terminator)
76  		while (data[end] != 0 && end < bufferEnd)
77  			end++;
78  
79  		// set to the end
80  		if (setToEnd && end != bufferEnd)
81  			buffer.position(end + 1); // skip 0 character
82  		
83  		return new String(data, start, end-start);
84  	}
85  
86  	/**
87  	 * Send error (exception) response.
88  	 * @param transport	transport to use.
89  	 * @param cid channel ID.
90  	 * @param errorStatus error status.
91  	 * @param previousHeader header of the request messages causing this error.
92  	 * @param message string message, can be <code>null</code>.
93  	 */
94  	protected void sendException(Transport transport, int cid, 
95  			CAStatus errorStatus, ByteBuffer previousHeader, String message)
96  	{
97  		try {
98  			// prepare for reading
99  			previousHeader.flip();
100 
101 			new ExceptionRequest(transport,	cid, errorStatus, previousHeader, message).submit();
102 		} catch (Throwable th) {
103 			context.getLogger().log(Level.WARNING, "Failed to send exception response for CID: " + cid, th);
104 		}
105 	}

109 	// type value codes
110 	private static final int DBR_TIME_OFFSET = 14;
111 	private static final int DBR_GR_OFFSET = 21;
112 	private static final int DBR_CTRL_OFFSET = 28;
113 
114 	/**
115 	 * Create appropriate DBR structure for reading (of PV type).
116 	 * Returned DBR structure will be the most complete one (CTRL usually)., PV array length.
117 	 * @param processVariable	process variable to be read.
118 	 * @return requested DBR.
119 	 */
120 	public static final DBR createDBRforReading(ProcessVariable processVariable)
121 	{
122 		return createDBRforReading(processVariable, processVariable.getDimensionSize(0));
123 	}
124 	
125 	/**
126 	 * Create appropriate DBR structure for reading (of PV type).
127 	 * Returned DBR structure will be the most complete one (CTRL usually).
128 	 * @param processVariable	process variable to be read.
129 	 * @param dataCount	requested data count (returned DBR will have this count).
130 	 * @return requested DBR.
131 	 */
132 	public static final DBR createDBRforReading(ProcessVariable processVariable, int dataCount)
133 	{
134 		return createDBRforReading(processVariable, (short)DBR_CTRL_OFFSET, dataCount);
135 	}
136 	
137 		/**
138 	 * Create appropriate DBR structure for reading (of PV type).
139 	 * @param processVariable	process variable to be read.
140 	 * @param dataType 	data type requested from client (not necessary equals to PV type)
141 	 * @param dataCount	requested data count (returned DBR will have this count).
142 	 * @return requested DBR.
143 	 */
144 	public static final DBR createDBRforReading(ProcessVariable processVariable, short dataType, int dataCount)
145 	{
146 
147 		// get native type (plain type) and convert it to TIME type (or GR, CTRL if requested)
148 		int offset = DBR_TIME_OFFSET;
149 		if (dataType >= DBR_GR_OFFSET)
150 			offset = (dataType >= DBR_CTRL_OFFSET) ? DBR_CTRL_OFFSET : DBR_GR_OFFSET;
151 
152 		// calculate new type
153 		DBRType type = processVariable.getType();
154 		int typeValue = type.getValue() + offset;
155 		
156 		DBR value;
157 		
158 		// special case for TIME_Double, TIME_Float
159 		// use an "artificial" TIME type with PRECISION
160 		// they are needed for string formatting
161 		if (typeValue == 20)
162 			value = new DBR_PRECISION_Double(dataCount);
163 		else if (typeValue == 16)
164 			value = new DBR_PRECISION_Float(dataCount);
165 		// special case to get the enum table, TIME_Enum -> LABELS_Enum
166 		// use an "artificial" DBR_TIME_LABELS type to support timestamping 
167 		else if (typeValue == 17 || typeValue == 24 || typeValue == 31 )
168 			value = new DBR_TIME_LABELS_Enum(dataCount);
169 		else
170 		{
171 			type = DBRType.forValue(typeValue);
172 	
173 			// create native DBR_{TIME, GR, CTRL}_<type> and set default values
174 			// use required count
175 			value = DBRFactory.create(type, dataCount);
176 		}
177 		
178 		// set undefined status
179 		STS dbrSts = (STS)value;
180 		dbrSts.setSeverity(Severity.INVALID_ALARM);
181 		dbrSts.setStatus(Status.UDF_ALARM);
182 		
183 		// initialize precision 
184 		if (value.isPRECSION())
185 			((PRECISION)value).setPrecision((short)-1);
186 
187 		return value;
188 	}
