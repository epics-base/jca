16  
17  import gov.aps.jca.CAException;
18  
19  import java.net.BindException;
20  import java.net.InetAddress;
21  import java.net.InetSocketAddress;
22  import java.net.SocketAddress;
23  import java.nio.channels.SelectionKey;
24  import java.nio.channels.ServerSocketChannel;
25  import java.nio.channels.SocketChannel;
26  import java.util.logging.Level;
27  import java.util.logging.Logger;
28  
29  import com.cosylab.epics.caj.impl.CAContext;
30  import com.cosylab.epics.caj.impl.reactor.ReactorHandler;
31  import com.cosylab.epics.caj.impl.reactor.lf.LeaderFollowersHandler;

33  /**
34   * Channel Access Server TCP acceptor.
35   * @author <a href="mailto:matej.sekoranjaATcosylab.com">Matej Sekoranja</a>
36   * @version $id$
37   */
38  public class CASAcceptor implements ReactorHandler {
39  
40  	// Get Logger
41  	private static final Logger logger = Logger.getLogger(CASAcceptor.class.getName());
42  	
43  	/**
44  	 * Context instance.
45  	 */
46  	private CAContext context;
47  	
48  	/**
49  	 * Bind server socket address.
50  	 */
51  	private InetSocketAddress bindAddress = null;
52  
53  	/**
54  	 * Server socket channel.
55  	 */
56  	private ServerSocketChannel serverSocketChannel = null;
57  	
58  	/**
59  	 * @param context
60  	 */
61  	public CASAcceptor(CAContext context, int port) throws CAException {
62  		this.context = context;
63  		
64  		int assignedPort = initialize(port);
65  		
66  		// update port, if dynamically assigned port is used
67  		if (assignedPort != port)
68  			((CAJServerContext)context).setTcpServerPort(assignedPort);
69  	}
70  
71  	/**
72  	 * Handle IO event.
73  	 * @see com.cosylab.epics.caj.impl.reactor.ReactorHandler#handleEvent(java.nio.channels.SelectionKey)
74  	 */
75  	public void handleEvent(SelectionKey key) {
76  		if (key.isValid() && key.isAcceptable())
77  		{
78  			try
79  			{
80  				ServerSocketChannel serverSocketChannel = (ServerSocketChannel)key.channel(); 
81  				SocketChannel socket = serverSocketChannel.accept();
82  				
83  				SocketAddress address = socket.socket().getRemoteSocketAddress();
84  				context.getLogger().finer("Accepted connection from CA client: " + address);
85  				
86  				// use non-blocking channel (no need for soTimeout)			
87  				socket.configureBlocking(false);
88  			
89  				// enable TCP_NODELAY (disable Nagle's algorithm)
90  				socket.socket().setTcpNoDelay(true);
91  				
92  				// enable TCP_KEEPALIVE
93  				socket.socket().setKeepAlive(true);
94  			
95  				// TODO tune buffer sizes?! Win32 defaults are 8k, which is OK
96  				//socket.socket().setReceiveBufferSize();
97  				//socket.socket().setSendBufferSize();
98  
99  				// create transport
100 				CASTransport transport = new CASTransport(context, socket);
101 				ReactorHandler handler = transport;
102 				if (context.getLeaderFollowersThreadPool() != null)
103 				    handler = new LeaderFollowersHandler(context.getReactor(), handler, context.getLeaderFollowersThreadPool());
104 				
105 				// register to reactor
106 				context.getReactor().register(socket, SelectionKey.OP_READ, handler);
107 				
108 				context.getLogger().finer("Serving to CA client: " + address);
109 
110 			} 
111 			catch (Throwable th)
112 			{
113 				logger.log(Level.SEVERE, "", th);
114 			}
115 		}
116 	}
117 
118 	/**
119 	 * Initialize connection acception. 
120 	 * @return port where server is listening
121 	 */
122 	private int initialize(int port) throws CAException
123 	{
124 		// specified bind address
125 		bindAddress = new InetSocketAddress(port);
126 
127 		int tryCount = 0;
128 		while (true)
129 		{
130 			tryCount++;
131 			
132 			try
133 			{
134 				context.getLogger().finer("Creating acceptor to " + bindAddress + ".");
135 	
136 				serverSocketChannel = ServerSocketChannel.open();
137 				serverSocketChannel.socket().bind(bindAddress);
138 				serverSocketChannel.configureBlocking(false);
139 				
140 				// update bind address, if dynamically port selection was used
141 				if (bindAddress.getPort() == 0)
142 				{
143 					bindAddress = new InetSocketAddress(serverSocketChannel.socket().getLocalPort());
144 					context.getLogger().info("Using dynamically assigned TCP port " + bindAddress.getPort() + ".");
145 				}
146 				
147 				// create handler
148 				ReactorHandler handler = this;
149 				if (context.getLeaderFollowersThreadPool() != null)
150 				    handler = new LeaderFollowersHandler(context.getReactor(), handler, context.getLeaderFollowersThreadPool());
151 				
152 				// register to reactor
153 				context.getReactor().register(serverSocketChannel, SelectionKey.OP_ACCEPT, handler);
154 	
155 				// rise level if port is assigned dynamically
156 				context.getLogger().finer("Accepting connections at " + bindAddress + ".");
157 
158 				// all OK, return
159 				return bindAddress.getPort();
160 			}
161 			catch (BindException be)
162 			{
163 				// failed to bind to specified bind address,
164 				// try to get port dynamically, but only once
165 				if (tryCount == 1)
166 				{
167 					context.getLogger().info("Configured TCP port " + port + " is unavailable, trying to assign it dynamically.");
168 					bindAddress = new InetSocketAddress(0);
169 				}
170 				else
171 				{
172 					throw new CAException("Failed to create acceptor to " + bindAddress, be);
173 				}				
174 			}
175 			catch (Throwable th)
176 			{
177 				throw new CAException("Failed to create acceptor to " + bindAddress, th);
178 			}
179 		}
180 		
181 	}
182 	
183 	/**
184 	 * Bind socket address.
185 	 * @return bind socket address, <code>null</code> if not binded.
186 	 */
187 	public InetAddress getBindAddress() 
188 	{
189 		return (serverSocketChannel != null) ? 
190 				serverSocketChannel.socket().getInetAddress() : null;
191 	}
192 	
193 	/**
194 	 * Destroy acceptor (stop listening).
195 	 */
196 	public void destroy()
197 	{
198 		if (serverSocketChannel != null)
199 		{
200 			context.getLogger().finer("Stopped accepting connections at " + bindAddress + ".");
201 
202 			context.getReactor().unregisterAndClose(serverSocketChannel);
203 		}
204 	}
205 	
206 }
