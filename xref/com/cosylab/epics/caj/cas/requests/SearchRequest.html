15  package com.cosylab.epics.caj.cas.requests;
16  
17  import java.net.InetAddress;
18  import java.nio.ByteBuffer;
19  
20  import com.cosylab.epics.caj.cas.CAJServerContext;
21  import com.cosylab.epics.caj.impl.CAConstants;
22  import com.cosylab.epics.caj.impl.Request;
23  import com.cosylab.epics.caj.impl.Transport;
24  import com.cosylab.epics.caj.impl.requests.AbstractCARequest;
25  import com.cosylab.epics.caj.util.InetAddressUtil;
26  
27  /**
28   * CA search response.
29   * @author <a href="mailto:matej.sekoranjaATcosylab.com">Matej Sekoranja</a>
30   * @version $id$
31   */
32  public class SearchRequest extends AbstractCARequest {
33  
34  	/**
35  	 * @param transport
36  	 */
37  	public SearchRequest(Transport transport, short clientMinorVersion, int cid) {
38  		super(transport);
39  
40  		// add minor version payload (aligned by 8)
41  		final int alignedMessageSize = CAConstants.CA_MESSAGE_HEADER_SIZE + 8;
42  		requestMessage = ByteBuffer.allocate(alignedMessageSize);
43  
44  		// set server IP address
45  		int serverIP = 0xFFFFFFFF;
46  		if (clientMinorVersion >= 8)
47  		{
48  			InetAddress serverAddress = ((CAJServerContext)(transport.getContext())).getServerInetAddress();
49  			if (serverAddress != null && !serverAddress.isAnyLocalAddress())
50  				serverIP = InetAddressUtil.ipv4AddressToInt(serverAddress);
51  		}
52  		
53  		requestMessage = insertCAHeader(transport, requestMessage,
54  										(short)6, (short)8, (short)transport.getContext().getServerPort(), 0,
55  										serverIP, cid);
56  		
57  		requestMessage.putShort(CAConstants.CAS_MINOR_PROTOCOL_REVISION);
58  		// clear rest of the message (security)
59  		requestMessage.putShort((short)0);
60  		requestMessage.putInt(0);
61  	}
62  
63  	/**
64  	 * @see com.cosylab.epics.caj.impl.Request#getPriority()
65  	 */
66  	public byte getPriority() {
67  		return Request.SEND_IMMEDIATELY_PRIORITY;
68  	}
69  
70  }
