15  package com.cosylab.epics.caj.cas.requests;
16  
17  import gov.aps.jca.CAStatus;
18  
19  import java.nio.ByteBuffer;
20  
21  import com.cosylab.epics.caj.impl.CAConstants;
22  import com.cosylab.epics.caj.impl.Request;
23  import com.cosylab.epics.caj.impl.Transport;
24  import com.cosylab.epics.caj.impl.requests.AbstractCARequest;
25  
26  /**
27   * CA exception (error) response.
28   * NOTE: extended header not supported (error messages are not so long).
29   * @author <a href="mailto:matej.sekoranjaATcosylab.com">Matej Sekoranja</a>
30   * @version $id$
31   */
32  public class ExceptionRequest extends AbstractCARequest {
33  
34      /**
35       * 
36       * @param transport to be used when sending.
37       * @param cid channel id 
38       * @param errorCode CAStatus
39       * @param previousHeader previous header
40       * @param errorMessage error message
41       */
42  	public ExceptionRequest(Transport transport, int cid, CAStatus errorCode,
43  							ByteBuffer previousHeader, String errorMessage) {
44  		super(transport);
45  
46  		// use default message if none provided
47  		int binaryMessageLength = 0;
48  		if (errorMessage != null)
49  			binaryMessageLength = errorMessage.length() + 1;
50  		 
51  		int alignedMessageSize = calculateAlignedSize(8, CAConstants.CA_MESSAGE_HEADER_SIZE + previousHeader.remaining() + binaryMessageLength);
52  		requestMessage = ByteBuffer.allocate(alignedMessageSize);
53  		requestMessage = insertCAHeader(transport, requestMessage,
54  										(short)11, (short)(alignedMessageSize - CAConstants.CA_MESSAGE_HEADER_SIZE),
55  										(short)0, (short)0,
56  										cid, errorCode.getStatusCode());
57  		
58  		// put previous header, i.e. header that caused an error
59  		requestMessage.put(previousHeader);								
60  		
61  		// put errorMessage
62  		if (binaryMessageLength > 0)
63  		{
64  			// append zero-terminated string and align message
65  			requestMessage.put(errorMessage.getBytes());
66  			requestMessage.put((byte)0);
67  		}
68  		
69  		requestMessage = alignBuffer(8, requestMessage);
70  	}
71  
72  	/**
73  	 * @see com.cosylab.epics.caj.impl.Request#getPriority()
74  	 */
75  	public byte getPriority() {
76  		return Request.SEND_IMMEDIATELY_PRIORITY;
77  	}
78  
79  }
