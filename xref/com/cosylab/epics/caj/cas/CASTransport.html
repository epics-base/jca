202 	/** 
203 	 * Close connection.
204 	 * @param forced	flag indicating if forced (e.g. forced disconnect) is required 
205 	 */
206 	public synchronized void close(boolean forced) {
207 
208 		// already closed check
209 		if (closed)
210 			return;
211 		closed = true;
212 
213 		// remove from registry, with priortiy CAConstants.CA_DEFAULT_PRIORITY (as registered!)
214 		context.getTransportRegistry().remove(socketAddress, CAConstants.CA_DEFAULT_PRIORITY);
215 
216 		// destroy all channels (and its monitors)
217 		destroyAllChannels();
218 		
219 		// shutdown event processor
220 		// wake up the processing thread
221 		synchronized (eventQueue) {
222 			eventQueue.notify();
223 		}
224 
225 		// flush first
226 		if (!forced)
227 			flushInternal();
228 		
229 		freeSendBuffers();
230 
231 		context.getLogger().finer("Connection to " + socketAddress + " closed.");
232 
233 		context.getReactor().unregisterAndClose(channel);
234 	}

236 	/**
237 	 * Destroy all channels.
238 	 */
239 	private void destroyAllChannels() {
240 
241 		ServerChannel[] channelsArray;
242 		synchronized (channels)
243 		{
244     		// resource allocation optimization
245     		if (channels.size() == 0)
246     			return;
247 
248     		channelsArray = new ServerChannel[channels.size()];
249 			channels.toArray(channelsArray);
250 
251 			channels.clear();
252 		}
253 
254 		context.getLogger().fine("Transport to " + socketAddress + " still has " + channelsArray.length + " channel(s) active and closing...");
255 		
256 		for (int i = 0; i < channelsArray.length; i++)
257 		{
258 			try
259 			{
260 				channelsArray[i].destroy();
261 			}
262 			catch (Throwable th)
263 			{
264 				logger.log(Level.SEVERE, "", th);
265 			}
266 		}
267 	}

308 	/**
309 	 * Process input (read) IO event.
310 	 */
311 	// TODO buffered read as in CAJ
312 	protected void processRead() {
313 		try
314 		{ 
315 			while (true)
316 			{
317 				if (closed) return;
318 				final ByteBuffer headerBuffer  = receiveBuffer[0];
319 				ByteBuffer payloadBuffer = receiveBuffer[1];
320 				
321 				// are we reading the header
322 				if (headerBuffer.hasRemaining())
323 				{
324 					if (channel.read(headerBuffer) < 0) {
325 						// error (disconnect, end-of-stream) detected
326 						close (true);
327 						return; 
328 					}
329 					
330 					// not done reading the header...
331 					if (headerBuffer.hasRemaining())
332 						break;
333 
334 					// peek for payload size (convert unsigned short to signed int) 
335 					int payloadSize = headerBuffer.getShort(2) & 0xFFFF;
336 					
337 					// extended message header
338 					if (payloadSize == 0xFFFF)
339 					{
340 						// already extended
341 						if (headerBuffer.limit() == CAConstants.CA_EXTENDED_MESSAGE_HEADER_SIZE)
342 						{
343 							// peek for extended payload
344 							payloadSize = headerBuffer.getInt(CAConstants.CA_MESSAGE_HEADER_SIZE);
345 							// if (payloadSize < 0) { /* this is way too much */ }
346 						}
347 						else
348 						{
349 							// extend to extended message header and re-read
350 							headerBuffer.limit(CAConstants.CA_EXTENDED_MESSAGE_HEADER_SIZE);
351 							continue;
352 						}
353 					}
354 
355 					// check payload buffer capacity
356 					if (payloadSize > payloadBuffer.capacity()) {
357 						receiveBuffer[1] = ByteBuffer.allocate(payloadSize);
358 						payloadBuffer = receiveBuffer[1];
359 					}
360 
361 					// reset payload buffer
362 					payloadBuffer.clear();
363 					payloadBuffer.limit(payloadSize);
364 				}
365 
366 				// are we reading the payload
367 				if (payloadBuffer.limit() == 0)
368 				{
369 					try
370 					{
371 						// prepare buffer for reading
372 						headerBuffer.flip();
373 
374 						// handle response					
375 						responseHandler.handleResponse(socketAddress, this, receiveBuffer);
376 					}
377 					catch (Throwable th)
378 					{
379 						// catch all bad code responses...	
380 						logger.log(Level.SEVERE, "", th);
381 					}
382 
383 					// reset header buffer
384 					headerBuffer.clear();
385 					headerBuffer.limit(CAConstants.CA_MESSAGE_HEADER_SIZE);
386 				}
387 				else if (payloadBuffer.hasRemaining())
388 				{
389 					channel.read(payloadBuffer);
390 
391 					// not done reading the payload...
392 					if (payloadBuffer.hasRemaining())
393 						break; 
394 						
395 					// prepare buffer for reading
396 					headerBuffer.flip();
397 
398 					// prepare buffer for reading
399 					payloadBuffer.flip();
400 					
401 					try
402 					{
403 						// handle response
404 						responseHandler.handleResponse(socketAddress, this, receiveBuffer);
405 					}
406 					catch (Throwable th)
407 					{
408 						// catch all bad code responses...	
409 						logger.log(Level.SEVERE, "", th);
410 					}
411 
412 					// reset header buffer
413 					headerBuffer.clear();
414 					headerBuffer.limit(CAConstants.CA_MESSAGE_HEADER_SIZE);
415 				}
416 			}
417 			
418 		} catch (IOException ioex) {
419 			// close connection
420 			close(true);
421 		} 
422 	}

432 	/**
433 	 * Send a buffer through the transport.
434 	 * NOTE: TCP sent buffer/sending has to be synchronized. 
435 	 * @param buffer	buffer to be sent
436 	 * @throws IOException 
437 	 */
438 	// TODO optimize !!!
439 	public void send(ByteBuffer buffer) throws IOException
440 	{
441 		synchronized (sendLock)
442 		{
443 			try
444 			{
445 				// prepare buffer
446 				buffer.flip();
447 
448 				final int SEND_BUFFER_LIMIT = 16000;
449 				int bufferLimit = buffer.limit();
450 
451 				//context.getLogger().finest("Sending " + bufferLimit + " bytes to " + socketAddress + ".");
452 
453 				// limit sending large buffers, split the into parts
454 				int parts = (buffer.limit()-1) / SEND_BUFFER_LIMIT + 1;
455 				for (int part = 1; part <= parts; part++)
456 				{
457 					if (parts > 1)
458 					{
459 						buffer.limit(Math.min(part * SEND_BUFFER_LIMIT, bufferLimit));
460 						context.getLogger().finest("[Parted] Sending (part " + part + "/" + parts + ") " + (buffer.limit()-buffer.position()) + " bytes to " + socketAddress + ".");
461 					}
462 					
463 					final int TRIES = 10;
464 					for (int tries = 0; /* tries <= TRIES */ ; tries++)
465 					{
466 						
467 						// send
468 						/*int bytesSent =*/ channel.write(buffer);
469 						// bytesSend == buffer.position(), so there is no need for flip()
470 						if (buffer.position() != buffer.limit())
471 						{
472 							if (tries == TRIES)
473 							{ 
474 								// TODO do sth ... and do not close transport
475 								context.getLogger().warning("Failed to send message to " + socketAddress + " - buffer full.");
476 								return;
477 							}
478 							
479 							// flush & wait for a while...
480 							context.getLogger().finest("Send buffer full for " + socketAddress + ", waiting...");
481 							channel.socket().getOutputStream().flush();
482 							try {
483 								Thread.sleep(10+tries*100);
484 							} catch (InterruptedException e) {
485 								// noop
486 							}
487 							continue;
488 						}
489 						else
490 							break;
491 					}
492 				
493 				}
494 				
495 			}
496 			catch (IOException ioex) 
497 			{
498 				// close connection
499 				close(true);
500 				throw ioex;
501 			}
502 		}
503 	}

506 	 * Flush task (to be executed by an thread pool).
507 	 */
508 	private Runnable flushTask =
509 		new Runnable() {
510 	        /**
511 	    	 * @see java.lang.Runnable#run()
512 	    	 */
513 	    	public void run() {
514 	    		flushInternal();
515 	    	}

518 	/**
519 	 * Flush send buffer.
520 	 * ... by enabling SelectionKey.OP_WRITE and process in reactor. 
521 	 * @see com.cosylab.epics.caj.impl.Transport#flush()
522 	 */
523 	public synchronized boolean flush()
524 	{
525 
526 	    // add to queue and flush
527 		synchronized(sendQueue)
528 		{
529 		    if (closed || sendBuffer == null)
530 		        return false;
531 		    
532 	        // noop check
533 	        if (sendBuffer.position() == 0)
534 	            return true;
535 	        else
536 	        {
537 	            // reuse old buffer
538 	            if (lastActiveSendBuffer != null &&
539 	                lastActiveSendBuffer.position()+sendBuffer.position() <= lastActiveSendBuffer.capacity())
540 	            {
541 	                sendBuffer.flip();
542 	                lastActiveSendBuffer.put(sendBuffer);
543 	                sendBuffer.clear();
544 	                return true;
545 	            }
546 	            
547 	            sendQueue.add(sendBuffer);
548 
549 	            // acquire new buffer
550 	            lastActiveSendBuffer = sendBuffer;
551 	    		sendBuffer = bufferAllocator.get();
552 	        }
553 	        
554 		    if (flushPending)
555 		        return true;
556 		    // NOTE: must be sure that code below will not fail
557 		    flushPending = true;
558 		}
559 	    
560 	    return spawnFlushing();
561 	}
562 	
563 	/**
564      * @return success flag.
565      */
566     private boolean spawnFlushing()
567     {
568         LeaderFollowersThreadPool lftp = context.getLeaderFollowersThreadPool();
569 	    if (lftp != null)
570 	    {
571 		    // reuse LF threadpool to do async flush
572 	        lftp.execute(flushTask);
573 	        return true;
574 	    }
575 	    else
576 	    {
577 		    // enable SelectionKey.OP_WRITE via reactor (this will also enable OP_READ, but its OK)
578 		    context.getReactor().setInterestOps(channel, SelectionKey.OP_READ | SelectionKey.OP_WRITE);
579 		    return true;
580 	    }
581     }

583     /**
584 	 * Flush send buffer (blocks until flushed).
585      * @return success flag.
586  	 * @see com.cosylab.epics.caj.impl.Transport#flush()
587 	 */
588 	public boolean flushInternal()
589 	{
590 	    // tricky closed check
591 	    if (sendBuffer == null)
592 	        return false;
593 	    
594 		try
595 		{
596 			while (sendQueue.size() > 0)
597 			{
598 				ByteBuffer buf;
599 				// dont want to block while sending...
600 				synchronized (sendQueue)
601 				{
602 					if (sendQueue.size() == 0)
603 						return true;
604 						
605 					buf = (ByteBuffer)sendQueue.removeFirst();
606 					// 'deactivate' lastActiveSendBuffer
607 					if (buf == lastActiveSendBuffer)
608 					    lastActiveSendBuffer = null;
609 				}
610 				
611 				try {
612 				    send(buf);
613 				}
614 				finally {
615 				    // return back to the cache
616 				    bufferAllocator.put(buf);
617 				}
618 			}
619 			
620 			return true;
621 		}
622 		catch (IOException ioex)
623 		{
624 			// close connection
625 			close(true);
626 			return false;
627 		}
628 		finally
629 		{
630 			synchronized (sendQueue)
631 			{
632 			    // ack
633 			    flushPending = false;
634 			    
635 			    // possible race condition check
636 				if (!closed && sendQueue.size() > 0)
637 				    spawnFlushing();
638 			}
639 		}
640 	}

642 	/**
643 	 * @see com.cosylab.epics.caj.impl.Transport#submit(com.cosylab.epics.caj.impl.Request)
644 	 */
645 	public void submit(Request requestMessage) throws IOException {
646 		ByteBuffer message = requestMessage.getRequestMessage();
647 			
648 		// empty message
649 		if (message.capacity() == 0)
650 			return;
651 				
652 		// send or enqueue
653 		if (requestMessage.getPriority() == Request.SEND_IMMEDIATELY_PRIORITY)
654 			send(message);
655 		else
656 		{
657 			message.flip();
658 
659 			synchronized (sendQueue) {
660 
661 			    if (sendBuffer == null)
662 				    throw new IllegalStateException("transport closed");
663 			    
664 			    // forced flush check
665 				if (message.limit()+sendBuffer.position() > sendBuffer.capacity())
666 				    flush();
667 				
668 				// TODO !!! check message size, it can exceed sendBuffer capacity
669 				sendBuffer.put(message);
670 			}
671 		}
672 	}

695 	/**
696 	 * CA server event process thread minimal priority.
697 	 */
698 	final static int CA_SERVER_THREAD_PRIORITY_MIN = Thread.NORM_PRIORITY - 1;
699 
700 	/**
701 	 * CA server event process thread maximal priority.
702 	 */
703 	final static int CA_SERVER_THREAD_PRIORITY_MAX = Thread.MAX_PRIORITY - 1;
704 	
705 	/**
706 	 * Calculate native thread priority from CA priortities.
707 	 * @param priority	CA priority
708 	 * @return native thread priority
709 	 */
710 	private final static int getNativeThreadPriority(short priority)
711 	{
712 		int nativePriority = priority - Channel.PRIORITY_MIN;
713 		nativePriority *= CA_SERVER_THREAD_PRIORITY_MAX - CA_SERVER_THREAD_PRIORITY_MIN;
714 		nativePriority /=  Channel.PRIORITY_MAX - Channel.PRIORITY_MIN;
715 		nativePriority += Thread.MIN_PRIORITY;
716 		return nativePriority;
717 	}

719 	/**
720 	 * Set transport priority.
721 	 */
722 	public void setPriority(short priority) {
723 
724 		// check for bounds
725 		if (priority < Channel.PRIORITY_MIN)
726 			priority = Channel.PRIORITY_MIN;
727 		else if (priority > Channel.PRIORITY_MAX)
728 			priority = Channel.PRIORITY_MAX;
729 		
730 		this.priority = priority;
731 		
732 		int nativePriority = getNativeThreadPriority(priority);
733 		if (nativePriority != Thread.currentThread().getPriority())
734 			Thread.currentThread().setPriority(nativePriority);
735 	}

905 	/**
906 	 * Process event.
907 	 * @param event	event to be processed.
908 	 * @return success flag.
909 	 */
910 	public boolean processEvents(Runnable event)
911 	{
912 		if (!processEvents || closed)
913 			return false;
914 
915 		// put into the queue to process
916         synchronized(eventQueue) {
917         	eventQueue.addLast(event);
918         	eventQueue.notify();
919         }
920 
921         return true;
922 	}
923 
924     /**
925 	 * Event dispatching is done here.
926 	 * @see java.lang.Runnable#run()
927 	 */
928 	public void run()
929 	{
930 		while (true)
931 		{
932 			Runnable event;
933 		
934 			synchronized (eventQueue)
935 			{
936 				// wait for new event ...
937 				while ((eventQueue.isEmpty() || !processEvents) && !closed) {
938 					try { 
939 						eventQueue.wait();
940 					} catch (InterruptedException ignored) { /* noop */ }
941 				}
942 
943 				// check if closed
944 				if (closed) {
945 					eventQueue.clear();
946 					return;
947 				}
948 
949 				event = (Runnable) eventQueue.removeFirst();
950 			}
951 
952 			// ... and process
953 			try {
954 				event.run();
955 			} catch (Throwable th) {
956 				logger.log(Level.SEVERE, "", th);
957 			}
958 		}
959 	}
