15  package com.cosylab.epics.caj.cas.util;
16  
17  import gov.aps.jca.CAException;
18  import gov.aps.jca.CAStatus;
19  import gov.aps.jca.cas.ProcessVariable;
20  import gov.aps.jca.cas.ProcessVariableEventCallback;
21  import gov.aps.jca.cas.ProcessVariableReadCallback;
22  import gov.aps.jca.cas.ProcessVariableWriteCallback;
23  import gov.aps.jca.dbr.DBR;
24  import gov.aps.jca.dbr.DBRType;
25  import gov.aps.jca.dbr.DBR_String;
26  import gov.aps.jca.dbr.DBR_TIME_String;
27  
28  /**
29   * Abstract convenient string process variable implementation. 
30   */
31  public abstract class StringProcessVariable extends ProcessVariable
32  {
33  	/**
34  	 * String PV constructor.
35  	 * @param name process variable name.
36  	 * @param eventCallback	event callback, can be <code>null</code>.
37  	 */
38  	public StringProcessVariable(String name, ProcessVariableEventCallback eventCallback)
39  	{
40  		super(name, eventCallback);
41  	}
42  
43  	/**
44  	 * Return <code>DBRType.STRING</code> type as native type.
45  	 * @see gov.aps.jca.cas.ProcessVariable#getType()
46  	 */
47  	public DBRType getType() {
48  		return DBRType.STRING;
49  	}
50  
51  	/**
52  	 * Casts <code>DBR</code> to <code>DBR_TIME_String</code> and
53  	 * delegates operation of reading a value to <code>readValue</code> method.
54  	 * @see gov.aps.jca.cas.ProcessVariable#read(gov.aps.jca.dbr.DBR, gov.aps.jca.cas.ProcessVariableReadCallback)
55  	 */
56  	public CAStatus read(DBR value, ProcessVariableReadCallback asyncReadCallback) throws CAException {
57  
58  		// for STRING type input DBR is always DBR_TIME_String
59  		DBR_TIME_String stringValue = (DBR_TIME_String)value;
60  
61  		// read value
62  		return readValue(stringValue, asyncReadCallback);
63  	}
64  
65  	/**
66  	 * Read value.
67  	 * Reference implementation:
68  	 * <pre>
69  	 * 	{
70  	 * 
71  	 * 		// for async. completion, return <code>null</code>,
72  	 * 		// set value (and status) to <code>enumValue</code> and
73  	 * 		// report completion using <code>asyncReadCallback</code> callback.
74  	 * 		// return null;
75  	 * 	
76  	 *      {@code
77  	 * 		// BEGIN optional (to override defaults) 
78  	 * 		
79  	 * 		// set status and severity
80  	 * 		value.setStatus(gov.aps.jca.dbr.Status);
81  	 * 		value.setSeverity(gov.aps.jca.dbr.Severity);
82  	 * 		
83  	 * 		// set timestamp
84  	 * 		value.setTimestamp(timestamp);
85  	 * 		
86  	 * 		// END optional (to override defaults)
87   	 * 		
88  	 * 		// set value to given DBR (example of copying value)
89  	 * 		// given DBR has already allocated an array of elements client has requested
90  	 * 		// it contains maximum number of elements to fill
91  	 * 		String[] arrayValue = value.getStringValue();
92  	 * 		int elementCount = Math.min(\<fromStringArray\>.length, arrayValue.length);
93  	 * 		System.arraycopy(\<fromStringArray\>, 0, arrayValue, 0, elementCount);
94  	 * 
95  	 * 		// return read completion status
96  	 * 		return CAStatus.NORMAL;
97  	 *      }
98  	 * 	} 
99  	 * 
100 	 * </pre>
101 	 * @see gov.aps.jca.cas.ProcessVariable#read(gov.aps.jca.dbr.DBR, gov.aps.jca.cas.ProcessVariableReadCallback)
102 	 */
103 	protected abstract CAStatus readValue(DBR_TIME_String value,
104 			ProcessVariableReadCallback asyncReadCallback) throws CAException;
105 		
106 	
107 	/**
108 	 * Casts <code>DBR</code> to <code>DBR_String</code> and delegates operation of writing a value to <code>writeValue</code> method.
109 	 * @see gov.aps.jca.cas.ProcessVariable#write(gov.aps.jca.dbr.DBR, gov.aps.jca.cas.ProcessVariableWriteCallback)
110 	 */
111 	public CAStatus write(DBR value, ProcessVariableWriteCallback asyncWriteCallback) throws CAException {
112 		
113 		// for STRING type input DBR is always DBR_String
114 		DBR_String stringValue = (DBR_String)value;
115 		
116 		return writeValue(stringValue, asyncWriteCallback);
117 	}
118 	
119     /**
120      * Write value. Reference implementation:
121      * 
122      * <pre>
123      * {
124      * 
125      *     // for async. completion, return <code>null</code>,
126      *     // set value (and status) from <code>enumValue</code>,
127      *     // notify if there is an interest and
128      *     // report completion using <code>asyncWriteCallback</code> callback.
129      *     // return null;
130      *     
131      *     {@code
132      *     // set value from given DBR here (scalar example)
133      *     this.value = ((DBR_String) value).getStringValue()[0];
134      * 
135      *     // notify, set appropirate Monitor mask (VALUE, LOG, ALARM)
136      *     if (status == CAStatus.NORMAL && interest) {
137      *         DBR monitorDBR = AbstractCASResponseHandler.createDBRforReading(this);
138      *         ((DBR_String) monitorDBR).getStringValue()[0] = this.value;
139      *         fillInStatusAndTime((TIME) monitorDBR);
140      * 
141      *         eventCallback.postEvent(Monitor.VALUE | Monitor.LOG, value);
142      *     }
143      * 
144      *     // return read completion status
145      *     return CAStatus.NORMAL;
146      *     }
147      * }
148      * </pre>
149      * 
150      * @see gov.aps.jca.cas.ProcessVariable#write(gov.aps.jca.dbr.DBR,
151      *      gov.aps.jca.cas.ProcessVariableWriteCallback)
152      */
153 	protected abstract CAStatus writeValue(DBR_String value,
154 			ProcessVariableWriteCallback asyncWriteCallback) throws CAException;
155 
156 }
