15  package com.cosylab.epics.caj.cas.util;
16  
17  import gov.aps.jca.cas.ProcessVariableEventCallback;
18  import gov.aps.jca.dbr.DBR;
19  import gov.aps.jca.dbr.PRECISION;
20  
21  /**
22   * Abstract convenient floating decimal process variable implementation.
23   * This implementation specializes <code>NumericProcessVariable</code> - adds precision support.
24   * Implementation does not handle status, severity and alarms.
25   */
26  public abstract class FloatingDecimalProcessVariable extends NumericProcessVariable 
27  {
28  	/**
29  	 * Floating decimal PV constructor.
30  	 * @param name process variable name.
31  	 * @param eventCallback	event callback, can be <code>null</code>.
32  	 */
33  	public FloatingDecimalProcessVariable(String name, ProcessVariableEventCallback eventCallback)
34  	{
35  		super(name, eventCallback);
36  	}
37  
38  	/**
39  	 * Get precision (number of digits after decimal point).
40  	 * Use <code>-1</code> value as undefined precision.
41  	 * @return precision.
42  	 */
43  	public abstract short getPrecision();
44  
45  	/**
46  	 * Set data (units, limits, ...) and precision to DBR.
47  	 * @param value DBR to fill-in.
48  	 */
49  	public void fillInDBR(DBR value)
50  	{
51  		super.fillInDBR(value);
52  		
53  		if (value.isPRECSION()) {
54  			// fill PRECISION
55  			PRECISION precision = (PRECISION)value;
56  			precision.setPrecision(getPrecision());
57  		}
58  	}
59  }
