19  /**
20   * <p>A hash map that uses primitive ints for the key instead of objects. Also Entry objects are reusable.</p>
21   * NOTE: this implementatio is not synced, client has to care of that.
22   *
23   * <p>This implemenation is based on original java implementation.</p>
24   * @author <a href="mailto:matej.sekoranjaATcosylab.com">Matej Sekoranja</a>
25   * @version $id$
26   * @see java.util.HashMap
27   */
28  public class IntHashMap {
29  
30      /**
31       * The hash table data.
32       */
33      private transient Entry table[];
34  
35      /**
36       * The total number of entries in the hash table.
37       */
38      private transient int count;
39  
40      /**
41       * The table is rehashed when its size exceeds this threshold.  (The
42       * value of this field is (int)(capacity * loadFactor).)
43       *
44       * @serial
45       */
46      private int threshold;
47  
48      /**
49       * The load factor for the hashtable.
50       *
51       * @serial
52       */
53      private float loadFactor;
54  
55      /**
56       * Entry object pool.
57       *
58       */
59      private transient EntryObjectPool pool;
60  
61      /**
62       * <p>Innerclass that acts as a datastructure to create a new entry in the
63       * table.</p>
64       */
65      private static class Entry {
66          int hash;
67          int key;
68          Object value;
69          Entry next;
70  
71          /**
72           * <p>Create a new entry with the given values.</p>
73           *
74           * @param hash The code used to hash the object with
75           * @param key The key used to enter this in the table
76           * @param value The value for this key
77           * @param next A reference to the next entry in the table
78           */
79          protected Entry(int hash, int key, Object value, Entry next) {
80          	initialize(hash, key, value, next);
81          }
82          
83          /**
84           * <p>Set given values.</p>
85           *
86           * @param hash The code used to hash the object with
87           * @param key The key used to enter this in the table
88           * @param value The value for this key
89           * @param next A reference to the next entry in the table
90           */
91          public void initialize(int hash, int key, Object value, Entry next) {
92              this.hash = hash;
93              this.key = key;
94              this.value = value;
95              this.next = next;
96          }
97  
98      }

100     /**
101      * <p>Object pool implemenation for Entry class.</p>
102      */
103     // TODO timed clean-up
104     private static class EntryObjectPool {
105     	
106     	/**
107     	 * Pool of reusable objects.
108     	 */
109     	// TODO get rid of class casts
110     	private ArrayList pool;
111     	private int lastPos = -1;
112     	
113     	/**
114     	 * Constructor.
115     	 * @param initialCapacity	initial capacity of the pool.
116     	 */
117     	public EntryObjectPool(int initialCapacity) {
118     		pool = new ArrayList(initialCapacity);
119     	}
120     	
121     	/**
122     	 * Get <code>Entry</code> object from the object pool.
123     	 * 
124          * @param hash The code used to hash the object with
125          * @param key The key used to enter this in the table
126          * @param value The value for this key
127          * @param next A reference to the next entry in the table
128     	 * @return	<code>Entry</code> instance.
129     	 */
130     	public Entry getEntry(int hash, int key, Object value, Entry next) {
131     		if (lastPos == -1) {
132     			return new Entry(hash, key, value, next);
133     		}
134     		else {
135     			Entry entry = (Entry)pool.remove(lastPos--);
136     			entry.initialize(hash, key, value, next);
137     			return entry;
138     		}
139     	}
140     	
141     	/**
142     	 * Put (return) <code>Entry</code> object to the object pool.
143     	 * @param entry entry to put. 
144     	 */ 
145     	public void putEntry(Entry entry) {
146     		// NOTE: be sure that entry.value and entry.next are null
147     		pool.add(entry);
148     		lastPos++;
149     	}
150     }

152     /**
153      * <p>Constructs a new, empty hashtable with a default capacity and load
154      * factor, which is <code>20</code> and <code>0.75</code> respectively.</p>
155      */
156     public IntHashMap() {
157         this(20, 0.75f);
158     }

160     /**
161      * <p>Constructs a new, empty hashtable with the specified initial capacity
162      * and default load factor, which is <code>0.75</code>.</p>
163      *
164      * @param  initialCapacity the initial capacity of the hashtable.
165      * @throws IllegalArgumentException if the initial capacity is less
166      *   than zero.
167      */
168     public IntHashMap(int initialCapacity) {
169         this(initialCapacity, 0.75f);
170     }
171 
172     /**
173      * <p>Constructs a new, empty hashtable with the specified initial
174      * capacity and the specified load factor.</p>
175      *
176      * @param initialCapacity the initial capacity of the hashtable.
177      * @param loadFactor the load factor of the hashtable.
178      * @throws IllegalArgumentException  if the initial capacity is less
179      *             than zero, or if the load factor is nonpositive.
180      */
181     public IntHashMap(int initialCapacity, float loadFactor) {
182         super();
183         if (initialCapacity < 0) {
184             throw new IllegalArgumentException("Illegal Capacity: " + initialCapacity);
185         }
186         if (loadFactor <= 0) {
187             throw new IllegalArgumentException("Illegal Load: " + loadFactor);
188         }
189         if (initialCapacity == 0) {
190             initialCapacity = 1;
191         }
192 
193         this.loadFactor = loadFactor;
194         table = new Entry[initialCapacity];
195         threshold = (int) (initialCapacity * loadFactor);
196         
197         pool = new EntryObjectPool(initialCapacity);
198     }

219     /**
220      * <p>Tests if some key maps into the specified value in this hashtable.
221      * This operation is more expensive than the <code>containsKey</code>
222      * method.</p>
223      *
224      * <p>Note that this method is identical in functionality to containsValue,
225      * (which is part of the Map interface in the collections framework).</p>
226      *
227      * @param      value   a value to search for.
228      * @return     <code>true</code> if and only if some key maps to the
229      *             <code>value</code> argument in this hashtable as
230      *             determined by the <tt>equals</tt> method;
231      *             <code>false</code> otherwise.
232      * @throws  NullPointerException  if the value is <code>null</code>.
233      * @see        #containsKey(int)
234      * @see        #containsValue(Object)
235      * @see        java.util.Map
236      */
237     public boolean contains(Object value) {
238         if (value == null) {
239             throw new NullPointerException();
240         }
241 
242         Entry tab[] = table;
243         for (int i = tab.length; i-- > 0;) {
244             for (Entry e = tab[i]; e != null; e = e.next) {
245                 if (e.value.equals(value)) {
246                     return true;
247                 }
248             }
249         }
250         return false;
251     }

254      * <p>Returns <code>true</code> if this HashMap maps one or more keys
255      * to this value.</p>
256      *
257      * <p>Note that this method is identical in functionality to contains
258      * (which predates the Map interface).</p>
259      *
260      * @param value value whose presence in this HashMap is to be tested.
261      * @return boolean <code>true</code> if the value is contained
262      * @see    java.util.Map
263      * @since JDK1.2
264      */
265     public boolean containsValue(Object value) {
266         return contains(value);
267     }

269     /**
270      * <p>Tests if the specified object is a key in this hashtable.</p>
271      *
272      * @param  key  possible key.
273      * @return <code>true</code> if and only if the specified object is a
274      *    key in this hashtable, as determined by the <tt>equals</tt>
275      *    method; <code>false</code> otherwise.
276      * @see #contains(Object)
277      */
278     public boolean containsKey(int key) {
279         Entry tab[] = table;
280         int hash = key;
281         int index = (hash & 0x7FFFFFFF) % tab.length;
282         for (Entry e = tab[index]; e != null; e = e.next) {
283             if (e.hash == hash) {
284                 return true;
285             }
286         }
287         return false;
288     }

290     /**
291      * <p>Returns the value to which the specified key is mapped in this map.</p>
292      *
293      * @param   key   a key in the hashtable.
294      * @return  the value to which the key is mapped in this hashtable;
295      *          <code>null</code> if the key is not mapped to any value in
296      *          this hashtable.
297      * @see     #put(int, Object)
298      */
299     public Object get(int key) {
300         Entry tab[] = table;
301         int hash = key;
302         int index = (hash & 0x7FFFFFFF) % tab.length;
303         for (Entry e = tab[index]; e != null; e = e.next) {
304             if (e.hash == hash) {
305                 return e.value;
306             }
307         }
308         return null;
309     }

311     /**
312      * <p>Increases the capacity of and internally reorganizes this
313      * hashtable, in order to accommodate and access its entries more
314      * efficiently.</p>
315      *
316      * <p>This method is called automatically when the number of keys
317      * in the hashtable exceeds this hashtable's capacity and load
318      * factor.</p>
319      */
320     protected void rehash() {
321         int oldCapacity = table.length;
322         Entry oldMap[] = table;
323 
324         int newCapacity = oldCapacity * 2 + 1;
325         Entry newMap[] = new Entry[newCapacity];
326 
327         threshold = (int) (newCapacity * loadFactor);
328         table = newMap;
329 
330         for (int i = oldCapacity; i-- > 0;) {
331             for (Entry old = oldMap[i]; old != null;) {
332                 Entry e = old;
333                 old = old.next;
334 
335                 int index = (e.hash & 0x7FFFFFFF) % newCapacity;
336                 e.next = newMap[index];
337                 newMap[index] = e;
338             }
339         }
340     }

342     /**
343      * <p>Maps the specified <code>key</code> to the specified
344      * <code>value</code> in this hashtable. The key cannot be
345      * <code>null</code>. </p>
346      *
347      * <p>The value can be retrieved by calling the <code>get</code> method
348      * with a key that is equal to the original key.</p>
349      *
350      * @param key     the hashtable key.
351      * @param value   the value.
352      * @return the previous value of the specified key in this hashtable,
353      *         or <code>null</code> if it did not have one.
354      * @throws  NullPointerException  if the key is <code>null</code>.
355      * @see     #get(int)
356      */
357     public Object put(int key, Object value) {
358         // Makes sure the key is not already in the hashtable.
359         Entry tab[] = table;
360         int hash = key;
361         int index = (hash & 0x7FFFFFFF) % tab.length;
362         for (Entry e = tab[index]; e != null; e = e.next) {
363             if (e.hash == hash) {
364                 Object old = e.value;
365                 e.value = value;
366                 return old;
367             }
368         }
369 
370         if (count >= threshold) {
371             // Rehash the table if the threshold is exceeded
372             rehash();
373 
374             tab = table;
375             index = (hash & 0x7FFFFFFF) % tab.length;
376         }
377 
378         // Creates the new entry.
379         //Entry e = new Entry(hash, key, value, tab[index]);
380         Entry e = pool.getEntry(hash, key, value, tab[index]);
381         tab[index] = e;
382         count++;
383         return null;
384     }

386     /**
387      * <p>Removes the key (and its corresponding value) from this
388      * hashtable.</p>
389      *
390      * <p>This method does nothing if the key is not present in the
391      * hashtable.</p>
392      *
393      * @param   key   the key that needs to be removed.
394      * @return  the value to which the key had been mapped in this hashtable,
395      *          or <code>null</code> if the key did not have a mapping.
396      */
397     public Object remove(int key) {
398         Entry tab[] = table;
399         int hash = key;
400         int index = (hash & 0x7FFFFFFF) % tab.length;
401         for (Entry e = tab[index], prev = null; e != null; prev = e, e = e.next) {
402             if (e.hash == hash) {
403                 if (prev != null) {
404                     prev.next = e.next;
405                 } else {
406                     tab[index] = e.next;
407                 }
408                 count--;
409                 Object oldValue = e.value;
410                 e.value = null;
411                 
412                 e.next = null;
413                 pool.putEntry(e);
414                 
415                 return oldValue;
416             }
417         }
418         return null;
419     }

421     /**
422      * <p>Clears this hashtable so that it contains no keys.</p>
423      */
424     public void clear() {
425         Entry tab[] = table;
426         for (int index = tab.length; --index >= 0;) {
427         	
428         	Entry e = tab[index];
429         	if (e != null) {
430         		e.value = null;
431         		e.next = null;
432         		pool.putEntry(e);
433         	}
434         	
435             tab[index] = null;
436         }
437         count = 0;
438     }

440     /**
441      * <p>Copies values to array. Note that array capacity has to be large enough.</p>
442      * @param arr	array to be filled, new instance returned if given not large enough or <code>null</code>.
443      * @return array of values.
444      */
445     public Object[] toArray(Object[] arr) {
446         if (arr == null || arr.length < count) {
447             arr = new Object[count];
448         }
449 
450         int pos = 0;
451         Entry tab[] = table;
452         for (int i = tab.length; i-- > 0;)
453             for (Entry e = tab[i]; e != null; e = e.next)
454             	arr[pos++] = e.value;
455 
456         return arr;
457     }
