1   package org.epics.jca;
2   
3   import com.cosylab.epics.caj.cas.CAJServerContext;
4   import com.cosylab.epics.caj.cas.util.DefaultServerImpl;
5   import gov.aps.jca.CAException;
6   import gov.aps.jca.JCALibrary;
7   import gov.aps.jca.cas.ProcessVariable;
8   
9   public class EmbeddedIoc {
10      private JCALibrary jca;
11      private DefaultServerImpl server;
12      private CAJServerContext context;
13  
14      public EmbeddedIoc() throws CAException {
15          this(0); // 0 means dynamically assign
16      }
17  
18      public EmbeddedIoc(int udpPort) throws CAException {
19          jca = JCALibrary.getInstance();
20  
21          server = new DefaultServerImpl();
22          context = new CAJServerContext();
23          context.setUdpServerPort(udpPort);
24          context.initialize(server);
25      }
26  
27      /**
28       * The address to the server's UDP pv name search socket as one would use in a client addr_list.
29       *
30       * @return The server's pv name search UDP socket address
31       */
32      public String getAddress() {
33          return "localhost:" + context.getUdpServerPort();
34      }
35  
36      public void registerPv(ProcessVariable pv) {
37          server.registerProcessVariable(pv);
38      }
39  
40      public ProcessVariable unregisterPv(String name) {
41          return server.unregisterProcessVariable(name);
42      }
43  
44      public void start() {
45          new Thread(() -> {
46              try {
47                  context.run(0);
48              } catch (CAException e) {
49                  e.printStackTrace();
50              }
51          }).start();
52      }
53  
54      public void stop() throws CAException {
55          context.destroy();
56      }
57  }
