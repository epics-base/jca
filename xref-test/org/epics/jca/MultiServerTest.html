19  public class MultiServerTest {
20      private EmbeddedIoc ioc1;
21      private EmbeddedIoc ioc2;
22      private MonitorClient client;
23      private String[] pvs;
24  
25      @Before
26      public void setup() throws CAException {
27          ioc1 = new EmbeddedIoc();
28          ioc2 = new EmbeddedIoc();
29          client = new MonitorClient();
30  
31          pvs = new String[]{"channel1", "channel2"};
32  
33          ioc1.registerPv(new MemoryProcessVariable("channel1", null, DBRType.STRING, new String[]{"Hello from channel1!"}));
34          ioc2.registerPv(new MemoryProcessVariable("channel2", null, DBRType.STRING, new String[]{"Hello from channel2!"}));
35  
36          DefaultConfiguration config = new DefaultConfiguration("config");
37          config.setAttribute("class", JCALibrary.CHANNEL_ACCESS_JAVA);
38          config.setAttribute("auto_addr_list", "NO");
39          config.setAttribute("addr_list", ioc1.getAddress() + " " + ioc2.getAddress());
40  
41          long pollWaitMillis = 1000;
42          long pendIOTimeoutMillis = 2000;
43  
44          ioc1.start();
45          ioc2.start();
46          client.start(config, pvs, pollWaitMillis, pendIOTimeoutMillis);
47      }
48  
49      @After
50      public void tearDown() throws CAException {
51          client.stop();
52          ioc1.stop();
53          ioc2.stop();
54      }
55  
56      @Test
57      public void basicTest() throws InterruptedException, CAException, TimeoutException {
58          List<MonitorEvent> events = client.poll(); // Grabs most recent update, if any
59  
60          int actualCount = events.size();
61          String actualC2Value = null;
62  
63          for(MonitorEvent event: events) {
64              CAJChannel channel = (CAJChannel)event.getSource();
65  
66              if("channel2".equals(channel.getName())) {
67                  DBR_String dbr = (DBR_String)event.getDBR();
68                  String[] strArray = ((gov.aps.jca.dbr.STRING) dbr).getStringValue();
69                  actualC2Value = strArray[0];
70              }
71          }
72  
73          Assert.assertEquals(2, actualCount);
74          Assert.assertEquals("Hello from channel2!", actualC2Value);
75      }
76  }
