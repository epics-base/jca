21  /** Example for a Channel Access name server
22   * 
23   *  A name server is a CA server that listens to PV searches.
24   *  It does not, however, host any PVs.
25   *  Instead, it has a list of _other_ CA servers like IOCs
26   *  which host the PVs, and the CA name server then provides
27   *  the _other_ CA server's address.
28   * 
29   *  The main purpose is reduction of CA search traffic.
30   *  Instead of having clients broadcast their requests,
31   *  they can directly contact the CA name server.
32   * 
33   *  Example usage:
34   * 
35   *  In one terminal, run an IOC database with a record named "ramp"
36   *  under a non-default UDP and TCP port:
37   *      export EPICS_CA_SERVER_PORT=9876
38   *      softIoc -d test/resources/ramp.db 
39   * 
40   *  In another terminal, check that
41   *      caget ramp
42   *  will NOT be able to connect because it searches by default
43   *  via UDP port 5064, while the IOC runs on 9876 (UDP and TCP).
44   * 
45   *  Now run this CANameServer on the same host,
46   *      java -cp target/classes:target/test-classes -DCAJ_DEBUG=true com.cosylab.epics.caj.cas.test.CANameServer
47   *  and try `caget ramp` again.
48   *  The client will reach the name server via UDP 5064.
49   *  The name server replies with 127.0.0.1, port 9876
50   *  to the search request and client can then reach the IOC.
51   * 
52   *  The CANameServer also supports searches via TCP:
53   *      export EPICS_CA_NAME_SERVERS=127.0.0.1
54   *      export EPICS_CA_AUTO_ADDR_LIST=no
55   *      caget ramp
56   */
57  public class CANameServer
58  {
59      public static void main(String[] args) throws Exception
60      {   // Log as much as possible
61          final Logger logger = Logger.getLogger("");
62          logger.setLevel(Level.ALL);
63          for (Handler handler : logger.getHandlers())
64              handler.setLevel(Level.ALL);
65          
66          // Example for a name server database using some hardcoded values.
67          final Map<String, InetSocketAddress> names = new HashMap<>();
68          names.put("ramp", new InetSocketAddress("127.0.0.1", 9876));
69          names.put("Fred", new InetSocketAddress("1.2.3.4", 4242));
70          names.put("Jane", new InetSocketAddress("5.6.7.8", 5757));
71                  
72          JCALibrary jca = JCALibrary.getInstance();        
73          Server server = new Server()
74          {
75              @Override
76              public ProcessVariableExistanceCompletion processVariableExistanceTest(String name,
77                                                                                     InetSocketAddress client,
78                                                                                     ProcessVariableExistanceCallback callback)
79                      throws CAException, IllegalArgumentException, IllegalStateException
80              {
81                  System.out.println("Client " + client + " searches for '" + name + "'");
82                  final InetSocketAddress addr = names.get(name);
83                  if (addr != null)
84                      return ProcessVariableExistanceCompletion.EXISTS_ELSEWHERE(addr);
85                  else
86                      return ProcessVariableExistanceCompletion.DOES_NOT_EXIST_HERE;
87              }
88  
89              @Override
90              public ProcessVariable processVariableAttach(String name,
91                                                           ProcessVariableEventCallback event,
92                                                           ProcessVariableAttachCallback attach)
93                      throws CAStatusException, IllegalArgumentException, IllegalStateException
94              {
95                  throw new CAStatusException(CAStatus.NOSUPPORT, "not supported");
96              }
97          };
98          
99          final ServerContext context = jca.createServerContext(JCALibrary.CHANNEL_ACCESS_SERVER_JAVA, server);
100         context.run(0);
101     }
102 }
