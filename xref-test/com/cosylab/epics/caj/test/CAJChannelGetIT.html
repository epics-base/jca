114     /**
115      * Simple get tests.

117     public void testGet() throws CAException, TimeoutException, InterruptedException {
118         DBR dbr = channel.get();
119         context.pendIO(3.0);
120         assertEquals(channel.getFieldType(), dbr.getType());
121         assertEquals(channel.getElementCount(), dbr.getCount());
122         double[] retVal = (double[]) dbr.getValue();
123         assertEquals(dbr.getCount(), retVal.length);
124         assertEquals(12.34, retVal[0], 0.0);
125 
126         GetListenerImpl listener = new GetListenerImpl();
127         channel.get(listener);
128         synchronized (listener) {
129             context.flushIO();
130             listener.wait(3000);
131         }
132         assertEquals(CAStatus.NORMAL, listener.status);
133         dbr = listener.value;
134         assertEquals(channel.getFieldType(), dbr.getType());
135         assertEquals(channel.getElementCount(), dbr.getCount());
136         retVal = (double[]) dbr.getValue();
137         assertEquals(dbr.getCount(), retVal.length);
138         assertEquals(12.34, retVal[0], 0.0);
139     }

175     private void checkValue(Object value, int count) {
176         if (count <= 0)
177             return;
178 
179         if (value instanceof byte[]) {
180             byte[] val = (byte[]) value;
181             assertEquals(12, val[0]);
182             for (int i = 2; i < count; i++)
183                 assertEquals(0, val[0]);
184         } else if (value instanceof double[]) {
185             double[] val = (double[]) value;
186             assertEquals(12.34, val[0], 0.0);
187             for (int i = 2; i < count; i++)
188                 assertEquals(0, val[0], 0.0);
189         } else if (value instanceof float[]) {
190             float[] val = (float[]) value;
191             assertEquals(12.34, val[0], 0.0001);
192             for (int i = 2; i < count; i++)
193                 assertEquals(0, val[0], 0.0);
194         } else if (value instanceof int[]) {
195             int[] val = (int[]) value;
196             assertEquals(12, val[0]);
197             for (int i = 2; i < count; i++)
198                 assertEquals(0, val[0]);
199         } else if (value instanceof short[]) {
200             short[] val = (short[]) value;
201             assertEquals(12, val[0]);
202             for (int i = 2; i < count; i++)
203                 assertEquals(0, val[0]);
204         } else if (value instanceof String[]) {
205             String[] val = (String[]) value;
206             assertEquals("12.34", val[0]);
207             for (int i = 2; i < count; i++)
208                 assertEquals("0", val[0]);
209         }
210     }

212     private void internalGetTest(DBRType type, int count) throws CAException, TimeoutException, InterruptedException {
213 
214         DBR dbr = channel.get(type, count);
215         context.pendIO(3.0);
216         checkDBR(type, count, dbr);
217 
218         GetListenerImpl listener = new GetListenerImpl();
219         channel.get(type, count, listener);
220         synchronized (listener) {
221             context.flushIO();
222             listener.wait(3000);
223         }
224         assertEquals(CAStatus.NORMAL, listener.status);
225         checkDBR(type, count, listener.value);
226     }

228     /**
229      * @param type
230      * @param count
231      * @param dbr
232      */
233     private void checkDBR(DBRType type, int count, DBR dbr) {
234         assertEquals(type, dbr.getType());
235         assertEquals(count, dbr.getCount());
236         assertEquals(count, Array.getLength(dbr.getValue()));
237 
238         // TODO tmp check
239         if (type.getValue() == DBR_PutAckS.TYPE.getValue() || type.getValue() == DBR_PutAckT.TYPE.getValue()) {
240             // what to do here...
241         } else
242             checkValue(dbr.getValue(), count);
243 
244         if (dbr instanceof STS) {
245             STS sts = (STS) dbr;
246             assertEquals(Status.NO_ALARM, sts.getStatus());
247             assertEquals(Severity.NO_ALARM, sts.getSeverity());
248         }
249         if (dbr instanceof TIME && !dbr.isGR()) {
250             TIME time = (TIME) dbr;
251             // 5 sec
252             long diff = Math.abs(new TimeStamp().secPastEpoch() - time.getTimeStamp().secPastEpoch());
253             assertTrue(diff < 100);
254         }
255         if (dbr instanceof GR) {
256             GR gr = (GR) dbr;
257             assertEquals(100, gr.getUpperDispLimit().doubleValue(), 0.0001);
258             assertEquals(5, gr.getUpperAlarmLimit().doubleValue(), 0.0001);
259             assertEquals(3, gr.getUpperWarningLimit().doubleValue(), 0.0001);
260             assertEquals(-3, gr.getLowerWarningLimit().doubleValue(), 0.0001);
261             assertEquals(-5, gr.getLowerAlarmLimit().doubleValue(), 0.0001);
262             assertEquals(-100, gr.getLowerDispLimit().doubleValue(), 0.0001);
263             assertEquals("units", gr.getUnits());
264         }
265         if (dbr instanceof CTRL) {
266             CTRL ctrl = (CTRL) dbr;
267             assertEquals(100, ctrl.getUpperCtrlLimit().doubleValue(), 0.0001);
268             assertEquals(-100, ctrl.getLowerCtrlLimit().doubleValue(), 0.0001);
269         }
270         if (dbr instanceof PRECISION) {
271             PRECISION prec = (PRECISION) dbr;
272             assertEquals(2, prec.getPrecision());
273         }
274     }

368     public void testEnum() throws CAException, TimeoutException, InterruptedException {
369 
370         channel.destroy();
371 
372         channel = context.createChannel("enum");
373         context.pendIO(5.0);
374 
375         final DBRType type = DBR_LABELS_Enum.TYPE;
376         final int count = 1;
377         final String[] expectedResult = new String[] { "zeroString", "oneString", "twoString", "threeString",
378                 "fourString", "fiveString", "sixString", "sevenString", "8s", "9s", "10s", "11s", "12s", "13s", "14s",
379                 "15s" };
380 
381         DBR dbr = channel.get(type, count);
382         context.pendIO(3.0);
383         assertTrue(dbr instanceof DBR_LABELS_Enum);
384 
385         DBR_LABELS_Enum label = (DBR_LABELS_Enum) dbr;
386         assertEquals(expectedResult.length, label.getLabels().length);
387         for (int i = 0; i < expectedResult.length; i++)
388             assertEquals(expectedResult[i], label.getLabels()[i]);
389 
390         GetListenerImpl listener = new GetListenerImpl();
391         channel.get(type, count, listener);
392         synchronized (listener) {
393             context.flushIO();
394             listener.wait(3000);
395         }
396         assertEquals(CAStatus.NORMAL, listener.status);
397         dbr = listener.value;
398         assertTrue(dbr instanceof DBR_LABELS_Enum);
399         label = (DBR_LABELS_Enum) dbr;
400         assertEquals(expectedResult.length, label.getLabels().length);
401         for (int i = 0; i < expectedResult.length; i++)
402             assertEquals(expectedResult[i], label.getLabels()[i]);
403     }
