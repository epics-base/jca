18  import gov.aps.jca.CAStatus;
19  import gov.aps.jca.Channel;
20  import gov.aps.jca.Context;
21  import gov.aps.jca.JCALibrary;
22  import gov.aps.jca.Monitor;
23  import gov.aps.jca.TimeoutException;
24  import gov.aps.jca.event.MonitorEvent;
25  import gov.aps.jca.event.MonitorListener;
26  
27  /**
28   * Simple basic usage test.
29   * @author <a href="mailto:matej.sekoranjaATcosylab.com">Matej Sekoranja</a>
30   * @version $id$
31   */
32  public class FlowControlTest {
33  
34  	/**
35       * Implementation of monitor listener.
36       */
37      private static class MonitorListenerImpl implements MonitorListener {
38          
39          /**
40           * @see gov.aps.jca.event.MonitorListener#monitorChanged(gov.aps.jca.event.MonitorEvent)
41           */
42          public void monitorChanged(MonitorEvent event)
43          {
44          	try {
45  				Thread.sleep(5);
46  			} catch (InterruptedException e) { /* noop */ }
47              // immediately print info
48              if (event.getStatus() == CAStatus.NORMAL)
49              {
50              	//System.out.println(((Channel)event.getSource()).getName());
51                  //event.getDBR().printInfo();
52              	//System.out.println();
53              }
54              else
55                  System.err.println("Monitor error: " + event.getStatus());
56          }
57      }
58      
59      /**
60       * JCA context.
61       */
62      private Context context = null;
63      
64      /**
65       * Initialize JCA context.
66       * @throws CAException	throws on any failure.
67       */
68      private void initialize() throws CAException {
69          
70  		// Get the JCALibrary instance.
71  		JCALibrary jca = JCALibrary.getInstance();
72  
73  		// Create a context with default configuration values.
74  		context = jca.createContext(JCALibrary.CHANNEL_ACCESS_JAVA);
75  
76  		// Display basic information about the context.
77          System.out.println(context.getVersion().getVersionString());
78          context.printInfo(); System.out.println();
79      }
80  
81      /**
82       * Destroy JCA context.
83       */
84      private void destroy() {
85          
86          try {
87  
88              // Destroy the context, check if never initialized.
89              if (context != null)
90                  context.destroy();
91              
92          } catch (Throwable th) {
93              th.printStackTrace();
94          }
95      }
96      
97  	/**
98  	 */
99  	public void execute() {
100 
101 		try {
102 		    
103 		    // initialize context
104 		    initialize();
105 
106 		    final int PHASE1_COUNT = 150;
107 		    final int PHASE2_COUNT = 199;
108 		    
109 			// Create the Channel to connect to the PV.
110 			Channel[] channels = new Channel[PHASE2_COUNT];
111 			for (int i = 0; i < PHASE1_COUNT; i++)
112 				channels[i] = context.createChannel("record" + i);
113 
114 			// Send the request and wait 5.0 seconds for the channel to connect to the PV.
115 			try
116 			{
117 				context.pendIO(5.0);
118 			} catch(TimeoutException te) {
119 				for (int i = 0; i < PHASE1_COUNT; i++)
120 					if (channels[i].getConnectionState() != Channel.CONNECTED)
121 						System.err.println("Failed to connect to record" + i);
122                 throw te;				
123 			}
124 			
125 			System.out.println("All connected.");
126 			
127 
128             /********************************************************************/ 
129             /***************************** Monitors *****************************/ 
130             /********************************************************************/ 
131             
132             System.out.println("\n------------------------------------------------\n");
133             System.out.println("Monitors...");
134             System.out.println();
135 
136             // Create a monitors
137 			for (int i = 0; i < PHASE1_COUNT; i++)
138 				channels[i].addMonitor(channels[i].getFieldType(), 4, Monitor.VALUE, new MonitorListenerImpl());
139 				//channels[i].addMonitor(Monitor.VALUE, new MonitorListenerImpl());
140             context.flushIO();
141 
142             // Sleep for 15 seconds (monitors will be printed out).
143             Thread.sleep(15000);
144 
145             System.out.println("Going phase 2..");
146             
147             // go go go phase2
148 			for (int i = PHASE1_COUNT; i < PHASE2_COUNT; i++)
149 				channels[i] = context.createChannel("record" + i);
150 
151 			// Send the request and wait 10.0 seconds for the channel to connect to the PV.
152 			try
153 			{
154 				context.pendIO(0.0);
155 			} catch(TimeoutException te) {
156 				for (int i = PHASE1_COUNT; i < PHASE2_COUNT; i++)
157 					if (channels[i].getConnectionState() != Channel.CONNECTED)
158 						System.err.println("Failed to connect to record" + i);
159                 throw te;				
160 			}
161 			
162 			System.out.println("Phase2: All connected.");
163 
164 			// do some work for a few seconds
165 			Thread.sleep(5000);
166 
167 			System.out.println("\n------------------------------------------------");
168             
169             /********************************************************************/ 
170             /********************************************************************/ 
171             /********************************************************************/ 
172 
173 			// Flush all pending requests...
174             context.flushIO();
175 
176 			System.out.println("Done.");
177 		} catch (Throwable th) {
178 			th.printStackTrace();
179 		}
180 		finally {
181 		    // always finalize
182 		    destroy();
183 		}
184 
185 	}
186 	
187 	
188 	/**
189 	 * Program entry point. 
190 	 * @param args	command-line arguments
191 	 */
192 	public static void main(String[] args) {
193 		// execute
194 		new FlowControlTest().execute();
195 	}
196 	
197 }
